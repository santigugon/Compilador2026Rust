import torch
import triton
import triton.language as tl

@triton.jit
def _permute_copy_kernel(
    input_ptr, 
    output_ptr, 
    input_shape, 
    output_shape, 
    input_strides, 
    output_strides, 
    num_elements: tl.constexpr,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < num_elements
    
    # Load input element
    input_offsets = offsets
    input_indices = tl.zeros((BLOCK,), dtype=tl.int32)
    
    # Convert linear index to multi-dimensional indices
    temp = input_offsets
    for i in range(len(input_shape) - 1, -1, -1):
        input_indices = tl.where(i == len(input_shape) - 1, 
                                temp % input_shape[i], 
                                input_indices)
        temp = temp // input_shape[i]
    
    # Calculate output index from input indices
    output_index = 0
    for i in range(len(output_shape)):
        dim = i
        output_index += input_indices[dim] * output_strides[i]
    
    # Load and store
    x = tl.load(input_ptr + input_offsets, mask=mask, other=0.0)
    tl.store(output_ptr + output_index, x, mask=mask)

def permute_copy(input, dims):
    # Validate dims
    if len(dims) != len(input.shape):
        raise ValueError("dims must have the same length as input.shape")
    
    # Create output tensor with correct shape
    output_shape = [input.shape[i] for i in dims]
    out = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    
    # Calculate number of elements
    num_elements = input.numel()
    
    # Calculate strides for input and output
    input_strides = list(input.stride())
    output_strides = [out.stride(i) for i in range(len(out.shape))]
    
    # Create shape arrays
    input_shape = list(input.shape)
    output_shape = list(out.shape)
    
    # Launch kernel
    BLOCK = 256
    grid = (triton.cdiv(num_elements, BLOCK),)
    
    # Create a temporary tensor for the kernel to work with
    # We'll use a flattened view for the kernel
    input_flat = input.view(-1)
    output_flat = out.view(-1)
    
    _permute_copy_kernel[grid](
        input_flat, 
        output_flat, 
        input_shape, 
        output_shape, 
        input_strides, 
        output_strides, 
        num_elements, 
        BLOCK=BLOCK
    )
    
    return out
