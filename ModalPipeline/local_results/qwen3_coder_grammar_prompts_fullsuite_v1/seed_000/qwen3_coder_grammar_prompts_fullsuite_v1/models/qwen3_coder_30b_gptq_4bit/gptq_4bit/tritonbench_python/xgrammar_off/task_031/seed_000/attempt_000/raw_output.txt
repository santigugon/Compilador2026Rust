import torch
import triton
import triton.language as tl

@triton.jit
def mean_kernel(
    input_ptr,
    output_ptr,
    n_rows,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
    keepdim: tl.constexpr
):
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    # Load input data for this row
    input_row = tl.load(input_ptr + row_idx * n_cols + tl.arange(0, BLOCK_SIZE), mask=tl.arange(0, BLOCK_SIZE) < n_cols)
    
    # Compute mean
    mean_val = tl.sum(input_row) / n_cols
    
    # Write output
    if keepdim:
        tl.store(output_ptr + row_idx, mean_val)
    else:
        tl.store(output_ptr + row_idx, mean_val)

def mean_triton(input, dim, keepdim=False, dtype=None, out=None):
    if isinstance(dim, int):
        dim = (dim,)
    
    # Handle dtype casting
    if dtype is not None:
        input = input.to(dtype)
    
    # Validate dimensions
    input_shape = input.shape
    if len(dim) == 0:
        raise ValueError("dim must not be empty")
    
    # Normalize negative dimensions
    normalized_dims = []
    for d in dim:
        if d < 0:
            d = len(input_shape) + d
        if d < 0 or d >= len(input_shape):
            raise ValueError(f"Dimension {d} is out of range")
        normalized_dims.append(d)
    
    # Check if all dimensions are valid
    for d in normalized_dims:
        if input_shape[d] == 0:
            raise ValueError(f"Dimension {d} has size 0")
    
    # Create output shape
    output_shape = list(input_shape)
    if keepdim:
        for d in normalized_dims:
            output_shape[d] = 1
    else:
        for d in sorted(normalized_dims, reverse=True):
            output_shape.pop(d)
    
    # Handle case where all dimensions are reduced
    if len(output_shape) == 0:
        output_shape = [1]
    
    # Create output tensor
    if out is not None:
        output = out
        if output.shape != tuple(output_shape):
            raise ValueError(f"Output tensor shape {output.shape} does not match expected shape {tuple(output_shape)}")
    else:
        output = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    
    # Handle reduction over multiple dimensions
    # For simplicity, we'll reduce along the last dimension first
    # and then handle the rest iteratively
    
    # Flatten the tensor for easier processing
    if len(normalized_dims) == 1:
        # Single dimension reduction
        reduced_dim = normalized_dims[0]
        n_rows = 1
        n_cols = input_shape[reduced_dim]
        
        # Calculate total elements for other dimensions
        for i, s in enumerate(input_shape):
            if i != reduced_dim:
                n_rows *= s
        
        # Create appropriate grid
        grid = (n_rows,)
        BLOCK_SIZE = 1024
        
        # Launch kernel
        mean_kernel[grid](
            input_ptr=input.data_ptr(),
            output_ptr=output.data_ptr(),
            n_rows=n_rows,
            n_cols=n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            keepdim=keepdim
        )
    else:
        # Multiple dimensions - this is a simplified implementation
        # In practice, this would require more complex handling
        # For now, we'll just reduce along the last dimension
        # and assume the rest is handled by PyTorch's native implementation
        # This is a placeholder for a more complete implementation
        raise NotImplementedError("Multiple dimension reduction not fully implemented in Triton version")
    
    return output
