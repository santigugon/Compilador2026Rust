import torch
import triton
import triton.language as tl
import math

@triton.jit
def _qr_solve_kernel(
    A_ptr, b_ptr, x_ptr,
    m: tl.constexpr, n: tl.constexpr, k: tl.constexpr,
    stride_A_row, stride_A_col,
    stride_b_row, stride_b_col,
    stride_x_row, stride_x_col,
    BLOCK: tl.constexpr
):
    # Compute Q^T * b
    for i in range(k):
        for j in range(n):
            sum = tl.zeros([BLOCK], dtype=tl.float32)
            for l in range(m):
                a_val = tl.load(A_ptr + l * stride_A_row + j * stride_A_col)
                b_val = tl.load(b_ptr + l * stride_b_row + i * stride_b_col)
                sum = sum + a_val * b_val
            tl.store(x_ptr + j * stride_x_row + i * stride_x_col, sum)

@triton.jit
def _back_substitution_kernel(
    R_ptr, Qb_ptr, x_ptr,
    m: tl.constexpr, n: tl.constexpr, k: tl.constexpr,
    stride_R_row, stride_R_col,
    stride_Qb_row, stride_Qb_col,
    stride_x_row, stride_x_col,
    BLOCK: tl.constexpr
):
    # Back substitution to solve R * x = Q^T * b
    for i in range(k):
        for j in range(n-1, -1, -1):
            sum = tl.zeros([BLOCK], dtype=tl.float32)
            for l in range(j+1, n):
                r_val = tl.load(R_ptr + j * stride_R_row + l * stride_R_col)
                x_val = tl.load(x_ptr + l * stride_x_row + i * stride_x_col)
                sum = sum + r_val * x_val
            q_val = tl.load(Qb_ptr + j * stride_Qb_row + i * stride_Qb_col)
            r_diag = tl.load(R_ptr + j * stride_R_row + j * stride_R_col)
            result = (q_val - sum) / r_diag
            tl.store(x_ptr + j * stride_x_row + i * stride_x_col, result)

def fused_qr_solve(A: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
    # QR decomposition using torch
    Q, R = torch.linalg.qr(A)
    
    # Compute Q^T * b
    Qb = torch.matmul(Q.transpose(0, 1), b)
    
    # Solve R * x = Q^T * b using back substitution
    # Create output tensor
    x = torch.empty(R.shape[1], b.shape[1], dtype=A.dtype, device=A.device)
    
    # Use Triton for back substitution
    m, n = R.shape
    k = b.shape[1]
    
    # Create a temporary tensor for Q^T * b
    Qb_torch = torch.empty(n, k, dtype=A.dtype, device=A.device)
    
    # Copy Q^T * b to the temporary tensor
    Qb_torch.copy_(Qb)
    
    # Launch back substitution kernel
    BLOCK = 32
    grid = (triton.cdiv(n, BLOCK), triton.cdiv(k, BLOCK))
    
    # Create strides for Triton kernel
    stride_R_row = R.stride(0)
    stride_R_col = R.stride(1)
    stride_Qb_row = Qb_torch.stride(0)
    stride_Qb_col = Qb_torch.stride(1)
    stride_x_row = x.stride(0)
    stride_x_col = x.stride(1)
    
    # Launch kernel
    _back_substitution_kernel[grid](
        R, Qb_torch, x,
        m, n, k,
        stride_R_row, stride_R_col,
        stride_Qb_row, stride_Qb_col,
        stride_x_row, stride_x_col,
        BLOCK=BLOCK
    )
    
    return x
