import torch
import triton
import triton.language as tl
import math

@triton.jit
def _lu_decompose_kernel(A_ptr, L_ptr, U_ptr, pivot_ptr, n: tl.constexpr, batch_size: tl.constexpr, BLOCK: tl.constexpr):
    # For each batch
    batch_id = tl.program_id(0)
    # For each row in the matrix
    for i in range(n):
        # Compute U[i, j] for j >= i
        for j in range(i, n):
            # Load A[i, j] for current batch
            a_val = tl.load(A_ptr + batch_id * n * n + i * n + j)
            # Accumulate from previous columns
            for k in range(i):
                a_val = a_val - tl.load(L_ptr + batch_id * n * n + i * n + k) * tl.load(U_ptr + batch_id * n * n + k * n + j)
            # Store U[i, j]
            tl.store(U_ptr + batch_id * n * n + i * n + j, a_val)
            
        # Compute L[i, j] for j < i
        for j in range(i):
            # Load U[j, j] for current batch
            ujj = tl.load(U_ptr + batch_id * n * n + j * n + j)
            # Load A[i, j] for current batch
            a_val = tl.load(A_ptr + batch_id * n * n + i * n + j)
            # Accumulate from previous rows
            for k in range(j):
                a_val = a_val - tl.load(L_ptr + batch_id * n * n + i * n + k) * tl.load(U_ptr + batch_id * n * n + k * n + j)
            # Compute L[i, j] = a_val / U[j, j]
            l_val = a_val / ujj
            # Store L[i, j]
            tl.store(L_ptr + batch_id * n * n + i * n + j, l_val)
            
        # Find pivot
        if i < n - 1:
            max_val = tl.abs(tl.load(U_ptr + batch_id * n * n + i * n + i))
            max_idx = i
            for k in range(i + 1, n):
                abs_val = tl.abs(tl.load(U_ptr + batch_id * n * n + k * n + i))
                if abs_val > max_val:
                    max_val = abs_val
                    max_idx = k
            # Store pivot
            tl.store(pivot_ptr + batch_id * n + i, max_idx)

@triton.jit
def _solve_triangular_kernel(L_ptr, U_ptr, b_ptr, x_ptr, n: tl.constexpr, batch_size: tl.constexpr, BLOCK: tl.constexpr):
    # For each batch
    batch_id = tl.program_id(0)
    # Forward substitution for L
    for i in range(n):
        # Load b[i] for current batch
        b_val = tl.load(b_ptr + batch_id * n + i)
        # Subtract L[i, j] * x[j] for j < i
        for j in range(i):
            b_val = b_val - tl.load(L_ptr + batch_id * n * n + i * n + j) * tl.load(x_ptr + batch_id * n + j)
        # Store x[i]
        tl.store(x_ptr + batch_id * n + i, b_val)
        
    # Backward substitution for U
    for i in range(n - 1, -1, -1):
        # Load x[i] for current batch
        x_val = tl.load(x_ptr + batch_id * n + i)
        # Subtract U[i, j] * x[j] for j > i
        for j in range(i + 1, n):
            x_val = x_val - tl.load(U_ptr + batch_id * n * n + i * n + j) * tl.load(x_ptr + batch_id * n + j)
        # Divide by U[i, i]
        uii = tl.load(U_ptr + batch_id * n * n + i * n + i)
        x_val = x_val / uii
        # Store x[i]
        tl.store(x_ptr + batch_id * n + i, x_val)

def invert_matrix_lu(A, *, pivot=True, out=None):
    if not torch.is_tensor(A):
        raise TypeError("A must be a tensor")
    
    if A.dim() < 2:
        raise ValueError("A must be at least 2D")
    
    if A.shape[-1] != A.shape[-2]:
        raise ValueError("A must be square")
    
    batch_dims = A.shape[:-2]
    n = A.shape[-1]
    
    # Handle scalar case
    if n == 1:
        out_tensor = torch.empty_like(A)
        out_tensor = 1.0 / A
        return out_tensor
    
    # Create output tensor
    if out is not None:
        out_tensor = out
    else:
        out_tensor = torch.empty_like(A)
    
    # Handle batched case
    if len(batch_dims) > 0:
        batch_size = 1
        for dim in batch_dims:
            batch_size *= dim
    else:
        batch_size = 1
    
    # Allocate memory for L, U, and pivot arrays
    L = torch.zeros_like(A)
    U = torch.zeros_like(A)
    pivot = torch.zeros((batch_size, n), dtype=torch.int32)
    
    # Perform LU decomposition
    block = 256
    grid = (batch_size,)
    
    # For small matrices, use a simpler approach
    if n <= 32:
        # Use PyTorch's native LU decomposition for small matrices
        for i in range(batch_size):
            if len(batch_dims) > 0:
                # Handle batched case
                batch_idx = tuple(slice(None) for _ in range(len(batch_dims)))
                A_batch = A[batch_idx]
            else:
                A_batch = A
            
            # Perform LU decomposition using PyTorch
            L_batch, U_batch, P_batch = torch.linalg.lu(A_batch)
            
            # Compute inverse using the LU decomposition
            identity = torch.eye(n, dtype=A.dtype, device=A.device)
            # Solve for each column of identity matrix
            for j in range(n):
                # Solve L * y = P * identity[:, j]
                y = torch.linalg.solve_triangular(L_batch, P_batch @ identity[:, j], lower=True)
                # Solve U * x = y
                x = torch.linalg.solve_triangular(U_batch, y, upper=True)
                # Store result
                out_tensor[batch_idx] = x
    else:
        # For larger matrices, use Triton kernels
        # Perform LU decomposition
        _lu_decompose_kernel[grid](A, L, U, pivot, n, batch_size, BLOCK=block)
        
        # Create identity matrix
        identity = torch.eye(n, dtype=A.dtype, device=A.device)
        identity = identity.expand(batch_size, n, n)
        
        # Solve for each column of identity matrix
        x = torch.zeros_like(identity)
        
        # Solve each system using triangular solver
        for j in range(n):
            # Copy identity column to b
            b = identity[:, :, j]
            # Solve L * y = b
            _solve_triangular_kernel[grid](L, U, b, x, n, batch_size, BLOCK=block)
            # Copy result to output
            out_tensor[:, :, j] = x[:, :, j]
    
    return out_tensor
