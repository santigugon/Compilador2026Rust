```python
import torch
import triton
import triton.language as tl

@triton.jit
def _gelu_kernel(x_ptr, out_ptr, n: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    # GELU approximation using tanh
    y = 0.5 * x * (1.0 + tl.tanh(x * 0.7978845608))  # 0.7978845608 is sqrt(2/pi)
    tl.store(out_ptr + offsets, y, mask=mask)

def gelu(input, approximate='none'):
    out = torch.empty_like(input)
    n = input.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    
    if approximate == 'none':
        # Use exact GELU computation
        _gelu_kernel[grid](input, out, n, BLOCK=block)
    else:
        # For approximate, we can use the same kernel since it's already an approximation
        _gelu_kernel[grid](input, out, n, BLOCK=block)
    
    return out
```
