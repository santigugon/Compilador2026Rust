import torch
import triton
import triton.language as tl
from typing import Optional, Union

@triton.jit
def logspace_kernel(
    start_ptr, end_ptr, base_ptr, output_ptr,
    steps: int,
    BLOCK_SIZE: int = 1024
):
    pid = tl.program_id(0)
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offsets < steps
    
    start_val = tl.load(start_ptr)
    end_val = tl.load(end_ptr)
    base_val = tl.load(base_ptr)
    
    # Compute logspace values
    # For each step i, compute base^(start + i * (end-start)/(steps-1))
    step_size = (end_val - start_val) / (steps - 1)
    indices = offsets * step_size + start_val
    values = tl.power(base_val, indices)
    
    tl.store(output_ptr + offsets, values, mask=mask)

def logspace(start: Union[float, torch.Tensor], 
             end: Union[float, torch.Tensor], 
             steps: int, 
             base: float = 10.0, 
             *, 
             out: Optional[torch.Tensor] = None, 
             dtype: Optional[torch.dtype] = None, 
             layout: torch.layout = torch.strided, 
             device: Optional[torch.device] = None, 
             requires_grad: bool = False) -> torch.Tensor:
    
    # Handle tensor inputs
    if isinstance(start, torch.Tensor):
        start = start.item()
    if isinstance(end, torch.Tensor):
        end = end.item()
    
    # Validate steps
    if steps <= 0:
        raise ValueError("steps must be positive")
    
    # Set default dtype if needed
    if dtype is None:
        dtype = torch.get_default_dtype()
    
    # Create output tensor
    if out is not None:
        result = out
    else:
        result = torch.empty(steps, dtype=dtype, device=device, layout=layout, requires_grad=requires_grad)
    
    # Prepare input tensors for Triton kernel
    start_tensor = torch.tensor(start, dtype=torch.float32, device=device)
    end_tensor = torch.tensor(end, dtype=torch.float32, device=device)
    base_tensor = torch.tensor(base, dtype=torch.float32, device=device)
    
    # Launch Triton kernel
    grid = (triton.cdiv(steps, 1024),)
    logspace_kernel[grid](
        start_tensor,
        end_tensor,
        base_tensor,
        result.data_ptr(),
        steps
    )
    
    return result
