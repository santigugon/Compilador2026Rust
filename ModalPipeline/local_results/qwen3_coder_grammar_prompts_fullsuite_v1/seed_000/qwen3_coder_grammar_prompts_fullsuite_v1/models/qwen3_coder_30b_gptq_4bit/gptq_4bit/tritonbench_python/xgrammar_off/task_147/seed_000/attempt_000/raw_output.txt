import torch
import triton
import triton.language as tl

@triton.jit
def _normalize_kernel(x_ptr, out_ptr, n_elements: tl.constexpr, dim: tl.constexpr, p_norm: tl.constexpr, eps: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    # Compute norm
    if p_norm == 2.0:
        norm = tl.sqrt(tl.sum(x * x, axis=dim) + eps)
    elif p_norm == 1.0:
        norm = tl.sum(tl.abs(x), axis=dim) + eps
    else:
        norm = tl.pow(tl.sum(tl.pow(tl.abs(x), p_norm), axis=dim) + eps, 1.0 / p_norm)
    # Normalize
    normalized = x / norm
    tl.store(out_ptr + offsets, normalized, mask=mask)

@triton.jit
def _pairwise_distance_kernel(x1_ptr, x2_ptr, out_ptr, n1: tl.constexpr, n2: tl.constexpr, p_norm: tl.constexpr, eps: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n1 * n2
    x1 = tl.load(x1_ptr + offsets, mask=mask, other=0.0)
    x2 = tl.load(x2_ptr + offsets, mask=mask, other=0.0)
    # Compute distance
    if p_norm == 2.0:
        diff = x1 - x2
        distance = tl.sqrt(tl.sum(diff * diff, axis=1) + eps)
    elif p_norm == 1.0:
        diff = x1 - x2
        distance = tl.sum(tl.abs(diff), axis=1) + eps
    else:
        diff = x1 - x2
        distance = tl.pow(tl.sum(tl.pow(tl.abs(diff), p_norm), axis=1) + eps, 1.0 / p_norm)
    tl.store(out_ptr + offsets, distance, mask=mask)

def fused_pairwise_distance_normalize(x1: torch.Tensor, x2: torch.Tensor, p_norm: float = 2.0, eps_norm: float = 1e-12, eps_distance: float = 1e-6, keepdim: bool = False) -> torch.Tensor:
    # Normalize both tensors
    x1_normalized = torch.empty_like(x1)
    x2_normalized = torch.empty_like(x2)
    
    n1 = x1.numel()
    n2 = x2.numel()
    
    # Determine the dimension to normalize along
    if len(x1.shape) == 1:
        dim = 0
    else:
        dim = len(x1.shape) - 1
    
    # Normalize x1
    block = 256
    grid = (triton.cdiv(n1, block),)
    _normalize_kernel[grid](x1, x1_normalized, n1, dim, p_norm, eps_norm, BLOCK=block)
    
    # Normalize x2
    grid = (triton.cdiv(n2, block),)
    _normalize_kernel[grid](x2, x2_normalized, n2, dim, p_norm, eps_norm, BLOCK=block)
    
    # Compute pairwise distance
    out = torch.empty((x1.shape[0], x2.shape[0]), dtype=torch.float32)
    n = x1.shape[0] * x2.shape[0]
    block = 256
    grid = (triton.cdiv(n, block),)
    _pairwise_distance_kernel[grid](x1_normalized, x2_normalized, out, n1, n2, p_norm, eps_distance, BLOCK=block)
    
    if keepdim:
        out = out.unsqueeze(-1)
    
    return out
