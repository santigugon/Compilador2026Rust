```python
import torch
import triton
import triton.language as tl

@triton.jit
def _cos_signbit_kernel(x_ptr, cos_ptr, signbit_ptr, n: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    cos_x = tl.cos(x)
    signbit = (cos_x < 0.0).to(tl.int8)
    tl.store(cos_ptr + offsets, cos_x, mask=mask)
    tl.store(signbit_ptr + offsets, signbit, mask=mask)

def cos_signbit(input: torch.Tensor) -> tuple[torch.Tensor, torch.Tensor]:
    out_cos = torch.empty_like(input)
    out_signbit = torch.empty(input.shape, dtype=torch.int8, device=input.device)
    n = input.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    _cos_signbit_kernel[grid](input, out_cos, out_signbit, n, BLOCK=block)
    return out_cos, out_signbit
```
