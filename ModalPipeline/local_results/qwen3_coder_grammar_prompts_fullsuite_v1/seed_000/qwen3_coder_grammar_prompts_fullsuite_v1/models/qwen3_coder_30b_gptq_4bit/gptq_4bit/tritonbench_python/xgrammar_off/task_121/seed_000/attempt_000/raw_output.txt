import torch
import triton
import triton.language as tl

@triton.jit
def _gelu_kernel(x_ptr, out_ptr, n: tl.constexpr, approximate: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    if approximate == 'none':
        # Exact GELU: x * 0.5 * (1 + erf(x / sqrt(2)))
        sqrt_2 = 1.4142135623730951
        erf_x = tl.erf(x / sqrt_2)
        y = x * 0.5 * (1.0 + erf_x)
    else:
        # Approximate GELU using tanh
        # GELU approx = x * 0.5 * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
        pi = 3.141592653589793
        sqrt_2_over_pi = 1.4142135623730951 / pi
        x_cubed = x * x * x
        tanh_arg = sqrt_2_over_pi * (x + 0.044715 * x_cubed)
        tanh_x = tl.tanh(tanh_arg)
        y = x * 0.5 * (1.0 + tanh_x)
    
    tl.store(out_ptr + offsets, y, mask=mask)

@triton.jit
def _min_kernel(x_ptr, out_ptr, n: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Initialize minimum value
    min_val = tl.full([BLOCK], float('inf'), dtype=tl.float32)
    
    # Reduce to find minimum
    for i in range(0, n, BLOCK):
        current_offsets = i + tl.arange(0, BLOCK)
        current_mask = current_offsets < n
        current_x = tl.load(x_ptr + current_offsets, mask=current_mask, other=0.0)
        min_val = tl.minimum(min_val, current_x)
    
    # Store the minimum value
    tl.store(out_ptr + pid, min_val[0], mask=pid < 1)

def min_gelu(input, dim=None, keepdim=False, approximate='none', out=None):
    # Apply GELU activation
    input = input.float()
    out_gelu = torch.empty_like(input)
    
    n = input.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    
    # Determine approximate method
    approx_method = 'none' if approximate == 'none' else 'tanh'
    
    _gelu_kernel[grid](input, out_gelu, n, approx_method, BLOCK=block)
    
    # If no dimension is specified, return minimum of all elements
    if dim is None:
        # Use PyTorch's min function for simplicity
        result = torch.min(out_gelu)
        if out is not None:
            out.copy_(result)
            return out
        return result
    
    # If dimension is specified, reduce along that dimension
    result = torch.min(out_gelu, dim=dim, keepdim=keepdim)
    
    if out is not None:
        out.copy_(result.values if hasattr(result, 'values') else result)
        return out
    
    return result.values if hasattr(result, 'values') else result
