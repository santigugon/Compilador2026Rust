import torch

@triton.jit
def affine_grid_kernel(
    theta_ptr,
    grid_ptr,
    N,
    H_out,
    W_out,
    H_in,
    W_in,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // (H_out * W_out)
    if batch_idx >= N:
        return
    out_idx = pid % (H_out * W_out)
    h_out = out_idx // W_out
    w_out = out_idx % W_out
    
    # Compute grid coordinates
    if not align_corners:
        x_out = (w_out + 0.5) / W_out * 2 - 1
        y_out = (h_out + 0.5) / H_out * 2 - 1
    else:
        x_out = w_out / (W_out - 1) * 2 - 1
        y_out = h_out / (H_out - 1) * 2 - 1
    
    # Apply affine transformation
    theta_00 = tl.load(theta_ptr + batch_idx * 6 + 0)
    theta_01 = tl.load(theta_ptr + batch_idx * 6 + 1)
    theta_02 = tl.load(theta_ptr + batch_idx * 6 + 2)
    theta_10 = tl.load(theta_ptr + batch_idx * 6 + 3)
    theta_11 = tl.load(theta_ptr + batch_idx * 6 + 4)
    theta_12 = tl.load(theta_ptr + batch_idx * 6 + 5)
    
    x_in = x_out * theta_00 + y_out * theta_01 + theta_02
    y_in = x_out * theta_10 + y_out * theta_11 + theta_12
    
    # Normalize to input image coordinates
    x_in = (x_in + 1) * (W_in - 1) / 2
    y_in = (y_in + 1) * (H_in - 1) / 2
    
    # Store grid values
    tl.store(grid_ptr + batch_idx * H_out * W_out * 2 + out_idx * 2 + 0, x_in)
    tl.store(grid_ptr + batch_idx * H_out * W_out * 2 + out_idx * 2 + 1, y_in)

@triton.jit
def grid_sample_kernel(
    input_ptr,
    grid_ptr,
    output_ptr,
    N,
    C,
    H_out,
    W_out,
    H_in,
    W_in,
    mode,
    padding_mode,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // (H_out * W_out * C)
    if batch_idx >= N:
        return
    out_idx = pid % (H_out * W_out * C)
    c = out_idx // (H_out * W_out)
    h_out = (out_idx % (H_out * W_out)) // W_out
    w_out = (out_idx % (H_out * W_out)) % W_out
    
    # Get grid coordinates
    x_in = tl.load(grid_ptr + batch_idx * H_out * W_out * 2 + h_out * W_out * 2 + w_out * 2 + 0)
    y_in = tl.load(grid_ptr + batch_idx * H_out * W_out * 2 + h_out * W_out * 2 + w_out * 2 + 1)
    
    # Handle padding modes
    if padding_mode == 0:  # zeros
        if x_in < 0 or x_in >= W_in or y_in < 0 or y_in >= H_in:
            tl.store(output_ptr + batch_idx * H_out * W_out * C + out_idx, 0.0)
            return
    elif padding_mode == 1:  # border
        x_in = tl.maximum(0.0, tl.minimum(x_in, W_in - 1.0))
        y_in = tl.maximum(0.0, tl.minimum(y_in, H_in - 1.0))
    elif padding_mode == 2:  # reflection
        x_in = tl.abs(x_in)
        y_in = tl.abs(y_in)
        x_in = tl.fmod(x_in, 2 * W_in)
        y_in = tl.fmod(y_in, 2 * H_in)
        x_in = tl.where(x_in > W_in, 2 * W_in - x_in, x_in)
        y_in = tl.where(y_in > H_in, 2 * H_in - y_in, y_in)
    
    # Interpolation
    if mode == 0:  # bilinear
        x_f = tl.floor(x_in)
        y_f = tl.floor(y_in)
        x_d = x_in - x_f
        y_d = y_in - y_f
        
        # Sample four points
        x0 = x_f
        x1 = x_f + 1
        y0 = y_f
        y1 = y_f + 1
        
        # Clamp coordinates
        x0 = tl.maximum(0.0, tl.minimum(x0, W_in - 1.0))
        x1 = tl.maximum(0.0, tl.minimum(x1, W_in - 1.0))
        y0 = tl.maximum(0.0, tl.minimum(y0, H_in - 1.0))
        y1 = tl.maximum(0.0, tl.minimum(y1, H_in - 1.0))
        
        # Get values
        v00 = tl.load(input_ptr + batch_idx * H_in * W_in * C + c * H_in * W_in + y0 * W_in + x0)
        v01 = tl.load(input_ptr + batch_idx * H_in * W_in * C + c * H_in * W_in + y0 * W_in + x1)
        v10 = tl.load(input_ptr + batch_idx * H_in * W_in * C + c * H_in * W_in + y1 * W_in + x0)
        v11 = tl.load(input_ptr + batch_idx * H_in * W_in * C + c * H_in * W_in + y1 * W_in + x1)
        
        # Interpolate
        v0 = v00 * (1 - x_d) + v01 * x_d
        v1 = v10 * (1 - x_d) + v11 * x_d
        result = v0 * (1 - y_d) + v1 * y_d
        tl.store(output_ptr + batch_idx * H_out * W_out * C + out_idx, result)
    elif mode == 1:  # nearest
        x_r = tl.round(x_in)
        y_r = tl.round(y_in)
        x_r = tl.maximum(0.0, tl.minimum(x_r, W_in - 1.0))
        y_r = tl.maximum(0.0, tl.minimum(y_r, H_in - 1.0))
        result = tl.load(input_ptr + batch_idx * H_in * W_in * C + c * H_in * W_in + y_r * W_in + x_r)
        tl.store(output_ptr + batch_idx * H_out * W_out * C + out_idx, result)
