import torch
import triton
import triton.language as tl

@triton.jit
def _tril_mm_and_scale_kernel(
    A_ptr, B_ptr, out_ptr,
    n: tl.constexpr, p: tl.constexpr,
    alpha: tl.constexpr, beta: tl.constexpr,
    stride_A_row, stride_A_col,
    stride_B_row, stride_B_col,
    stride_out_row, stride_out_col,
    BLOCK: tl.constexpr
):
    # Compute output matrix C = alpha * tril(A) @ B + beta * C
    pid = tl.program_id(0)
    pid_j = tl.program_id(1)
    
    # Each block computes one row of the output matrix
    row = pid * BLOCK
    col = pid_j * BLOCK
    
    # Initialize accumulator
    acc = tl.zeros((BLOCK, BLOCK), dtype=tl.float32)
    
    # Loop over the K dimension (columns of A, rows of B)
    for k in range(0, n, BLOCK):
        # Load A block (lower triangular part)
        a_offsets = row * stride_A_row + k * stride_A_col
        a_mask = (row + tl.arange(0, BLOCK)[:, None] < n) & (k + tl.arange(0, BLOCK)[None, :] < n)
        a_block = tl.load(A_ptr + a_offsets, mask=a_mask, other=0.0)
        
        # Apply lower triangular mask
        a_block = tl.where(
            tl.arange(0, BLOCK)[:, None] >= tl.arange(0, BLOCK)[None, :],
            a_block,
            0.0
        )
        
        # Load B block
        b_offsets = k * stride_B_row + col * stride_B_col
        b_mask = (k + tl.arange(0, BLOCK)[:, None] < n) & (col + tl.arange(0, BLOCK)[None, :] < p)
        b_block = tl.load(B_ptr + b_offsets, mask=b_mask, other=0.0)
        
        # Accumulate
        acc += tl.dot(a_block, b_block)
    
    # Scale by alpha
    acc *= alpha
    
    # Store result
    out_offsets = row * stride_out_row + col * stride_out_col
    out_mask = (row + tl.arange(0, BLOCK)[:, None] < n) & (col + tl.arange(0, BLOCK)[None, :] < p)
    tl.store(out_ptr + out_offsets, acc, mask=out_mask)

def tril_mm_and_scale(A: torch.Tensor, B: torch.Tensor, alpha: float, beta: float) -> torch.Tensor:
    # Ensure inputs are 2D
    assert A.dim() == 2, "A must be a 2D tensor"
    assert B.dim() == 2, "B must be a 2D tensor"
    assert A.shape[0] == A.shape[1], "A must be square"
    assert A.shape[1] == B.shape[0], "A and B dimensions must be compatible for matrix multiplication"
    
    n, p = A.shape[0], B.shape[1]
    
    # Create output tensor
    out = torch.empty(n, p, dtype=torch.float32, device=A.device)
    
    # Define block size
    BLOCK = 32
    
    # Calculate grid dimensions
    grid = (triton.cdiv(n, BLOCK), triton.cdiv(p, BLOCK))
    
    # Launch kernel
    _tril_mm_and_scale_kernel[grid](
        A, B, out,
        n, p,
        alpha, beta,
        A.stride(0), A.stride(1),
        B.stride(0), B.stride(1),
        out.stride(0), out.stride(1),
        BLOCK=BLOCK
    )
    
    # Scale by beta
    out *= beta
    
    return out
