import torch
import triton
import triton.language as tl

@triton.jit
def _batch_norm_kernel(
    input_ptr, 
    weight_ptr, 
    bias_ptr, 
    running_mean_ptr, 
    running_var_ptr, 
    output_ptr,
    n_features: tl.constexpr,
    n_batch: tl.constexpr,
    eps: tl.constexpr,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    batch_offset = pid * BLOCK
    offsets = batch_offset + tl.arange(0, BLOCK)
    mask = offsets < n_batch
    
    # Load input data
    input = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    
    # Load weight and bias if they exist
    weight = tl.load(weight_ptr + offsets, mask=mask, other=1.0)
    bias = tl.load(bias_ptr + offsets, mask=mask, other=0.0)
    
    # Compute batch normalization
    mean = tl.sum(input) / n_features
    var = tl.sum((input - mean) * (input - mean)) / n_features
    
    # Use running statistics if not training
    if not training:
        mean = tl.load(running_mean_ptr + offsets, mask=mask, other=0.0)
        var = tl.load(running_var_ptr + offsets, mask=mask, other=0.0)
    
    # Normalize and scale
    normalized = (input - mean) / tl.sqrt(var + eps)
    output = normalized * weight + bias
    
    # Store output
    tl.store(output_ptr + offsets, output, mask=mask)

def batch_norm(input, running_mean, running_var, weight=None, bias=None, training=False, momentum=0.1, eps=1e-05):
    # Handle scalar weight and bias
    if weight is not None and not torch.is_tensor(weight):
        weight = torch.tensor(weight, device=input.device, dtype=input.dtype)
    if bias is not None and not torch.is_tensor(bias):
        bias = torch.tensor(bias, device=input.device, dtype=input.dtype)
    
    # Ensure weight and bias are tensors
    if weight is None:
        weight = torch.ones(input.shape[1], device=input.device, dtype=input.dtype)
    if bias is None:
        bias = torch.zeros(input.shape[1], device=input.device, dtype=input.dtype)
    
    # Ensure running_mean and running_var are tensors
    if running_mean is None:
        running_mean = torch.zeros(input.shape[1], device=input.device, dtype=input.dtype)
    if running_var is None:
        running_var = torch.ones(input.shape[1], device=input.device, dtype=input.dtype)
    
    # Create output tensor
    out = torch.empty_like(input)
    
    # Get dimensions
    n_batch = input.shape[0]
    n_features = input.shape[1]
    
    # Set block size
    BLOCK = 256
    
    # Launch kernel
    grid = (triton.cdiv(n_batch, BLOCK),)
    
    # For simplicity, we'll use a single kernel for both training and inference
    # In practice, you might want separate kernels for training and inference
    _batch_norm_kernel[grid](
        input, 
        weight, 
        bias, 
        running_mean, 
        running_var, 
        out,
        n_features,
        n_batch,
        eps,
        BLOCK=BLOCK
    )
    
    return out
