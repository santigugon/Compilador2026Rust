def _ldl_decompose_kernel(A_ptr, L_ptr, D_ptr, n: tl.constexpr, batch_size: tl.constexpr, BLOCK: tl.constexpr):

    pid = tl.program_id(0)

    row_col = pid % (n * n)

    col = row_col % n

        return

    pid = tl.program_id(0)
    batch_idx = pid // n
    row = pid % n
    
    if batch_idx >= batch_size:
        return

    y = tl.load(b_ptr + batch_idx * n + row)

            l_val = tl.load(L_ptr + batch_idx * n * n + row * n + k)

            x_val -= l_kj * tl.load(x_ptr + batch_idx * n + k)
