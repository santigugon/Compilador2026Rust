import triton
import triton.language as tl
import torch
from typing import Union, Tuple

@triton.jit
def sigmoid_adaptive_avg_pool2d_kernel(
    input_ptr, 
    output_ptr, 
    input_row_stride, 
    input_col_stride,
    output_row_stride, 
    output_col_stride,
    input_height, 
    input_width, 
    output_height, 
    output_width,
    num_elements,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    if pid >= num_elements:
        return
    
    output_row = pid // output_width
    output_col = pid % output_width
    
    if output_row >= output_height or output_col >= output_width:
        return
    
    start_h = (output_row * input_height) // output_height
    end_h = ((output_row + 1) * input_height + output_height - 1) // output_height
    start_w = (output_col * input_width) // output_width
    end_w = ((output_col + 1) * input_width + output_width - 1) // output_width
    
    h_range = end_h - start_h
    w_range = end_w - start_w
    
    if h_range == 0 or w_range == 0:
        return
    
    sum_val = tl.zeros([1], dtype=tl.float32)
    for h in range(start_h, end_h):
        for w in range(start_w, end_w):
            input_idx = h * input_row_stride + w * input_col_stride
            sum_val += tl.load(input_ptr + input_idx)
    
    avg_val = sum_val / (h_range * w_range)
    
    output_idx = output_row * output_row_stride + output_col * output_col_stride
    tl.store(output_ptr + output_idx, 1.0 / (1.0 + tl.exp(-avg_val)))

def sigmoid_adaptive_avg_pool2d(input: torch.Tensor, output_size: Union[int, Tuple[int, int]]) -> torch.Tensor:
    if isinstance(output_size, int):
        output_height = output_size
        output_width = output_size
    else:
        output_height, output_width = output_size
    
    input_height, input_width = input.shape[-2], input.shape[-1]
    
    output = torch.empty(input.shape[:-2] + (output_height, output_width), dtype=torch.float32, device=input.device)
    
    num_elements = output_height * output_width
    BLOCK_SIZE = 1024
    
    grid = (triton.cdiv(num_elements, BLOCK_SIZE),)
    
    sigmoid_adaptive_avg_pool2d_kernel[grid](
        input,
        output,
        input.stride(-2),
        input.stride(-1),
        output.stride(-2),
        output.stride(-1),
        input_height,
        input_width,
        output_height,
        output_width,
        num_elements,
        BLOCK_SIZE
    )
    
    return output
