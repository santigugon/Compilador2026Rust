import torch
import triton
import triton.language as tl

@triton.jit
def _det_lu_kernel(A_ptr, out_ptr, n, batch_size, BLOCK_SIZE: tl.constexpr):
    batch_idx = tl.program_id(0)
    if batch_idx >= batch_size:
        return
    
    # Load matrix A for this batch
    A_block = tl.load(A_ptr + batch_idx * n * n + tl.arange(0, n)[:, None] * n + tl.arange(0, n)[None, :])
    
    # Initialize U matrix and pivot tracking
    U = tl.zeros((n, n), dtype=tl.float32)
    pivot = tl.zeros((n,), dtype=tl.int32)
    
    # Copy A to U
    for i in range(n):
        for j in range(n):
            U[i, j] = A_block[i, j]
    
    # LU decomposition with partial pivoting
    sign = 1.0
    for k in range(n):
        # Find pivot
        max_val = tl.abs(U[k, k])
        pivot_idx = k
        for i in range(k + 1, n):
            if tl.abs(U[i, k]) > max_val:
                max_val = tl.abs(U[i, k])
                pivot_idx = i
        
        # Swap rows if needed
        if pivot_idx != k:
            for j in range(n):
                U[k, j], U[pivot_idx, j] = U[pivot_idx, j], U[k, j]
            sign *= -1.0
        
        # Check for singular matrix
        if abs(U[k, k]) < 1e-12:
            sign = 0.0
            break
        
        # Perform elimination
        for i in range(k + 1, n):
            factor = U[i, k] / U[k, k]
            for j in range(k + 1, n):
                U[i, j] -= factor * U[k, j]
    
    # Compute determinant as product of diagonal elements
    det = sign
    for i in range(n):
        det *= U[i, i]
    
    # Store result
    tl.store(out_ptr + batch_idx, det)

def determinant_lu(A, *, pivot=True, out=None):
    if A.dim() < 2:
        raise ValueError("Input tensor must have at least 2 dimensions")
    
    batch_dims = A.shape[:-2]
    n = A.shape[-1]
    
    if A.shape[-2] != n:
        raise ValueError("Input tensor must have square matrices")
    
    batch_size = 1
    for dim in batch_dims:
        batch_size *= dim
    
    if out is None:
        out = torch.empty(batch_dims, dtype=torch.float32, device=A.device)
    
    if out.shape != batch_dims:
        raise ValueError("Output tensor shape does not match batch dimensions")
    
    # Launch kernel
    grid = (batch_size, 1, 1)
    _det_lu_kernel[grid](A, out, n, batch_size, BLOCK_SIZE=32)
    
    return out
