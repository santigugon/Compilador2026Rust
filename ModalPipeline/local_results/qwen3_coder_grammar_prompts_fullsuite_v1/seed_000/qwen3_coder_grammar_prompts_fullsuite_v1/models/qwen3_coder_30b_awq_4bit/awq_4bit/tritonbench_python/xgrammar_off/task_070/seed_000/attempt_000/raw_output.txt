import torch
import triton
import triton.language as tl

@triton.jit
def _fused_add_mul_groupnorm_kernel(
    x_ptr, y_ptr, weight_ptr, bias_ptr, out_ptr,
    n_elements, num_groups, channels, groups, eps,
    x_stride, y_stride, weight_stride, bias_stride, out_stride,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    # Load inputs
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    y = tl.load(y_ptr + offsets, mask=mask, other=0.0)
    
    # Element-wise addition and multiplication
    xy = x + y
    xy = xy * y  # y is used as multiplier
    
    # Reshape for group normalization
    # For simplicity, we'll compute group norm on the flattened tensor
    # and reshape back to original shape
    xy_reshaped = xy.reshape(groups, channels, -1)
    
    # Compute mean and variance for group normalization
    mean = tl.sum(xy_reshaped, axis=2, keepdims=True) / (channels * xy_reshaped.shape[2])
    var = tl.sum((xy_reshaped - mean)**2, axis=2, keepdims=True) / (channels * xy_reshaped.shape[2])
    
    # Normalize
    normalized = (xy_reshaped - mean) / tl.sqrt(var + eps)
    
    # Apply weight and bias
    weight = tl.load(weight_ptr + tl.arange(0, channels), mask=tl.arange(0, channels) < channels, other=0.0)
    bias = tl.load(bias_ptr + tl.arange(0, channels), mask=tl.arange(0, channels) < channels, other=0.0)
    
    # Expand weight and bias to match the group structure
    weight_expanded = weight.reshape(1, channels, 1)
    bias_expanded = bias.reshape(1, channels, 1)
    
    # Apply normalization
    normalized = normalized * weight_expanded + bias_expanded
    
    # Reshape back to original shape
    out = normalized.reshape(-1)
    
    # Store result
    tl.store(out_ptr + offsets, out, mask=mask)

def fused_add_mul_groupnorm(input1, input2, weight, bias, num_groups, eps=1e-5, *, out=None):
    # Validate inputs
    if input1.shape != input2.shape:
        # Check if they are broadcastable
        try:
            torch.broadcast_tensors(input1, input2)
        except RuntimeError:
            raise ValueError("input1 and input2 must be broadcastable")
    
    # Ensure input1 and input2 have the same shape
    if input1.shape != input2.shape:
        # Broadcast input2 to match input1 shape
        input2 = input2.expand(input1.shape)
    
    # Validate weight and bias shapes
    if weight.shape != bias.shape:
        raise ValueError("weight and bias must have the same shape")
    
    # Validate num_groups
    if num_groups <= 0:
        raise ValueError("num_groups must be positive")
    
    # Get output tensor
    if out is None:
        out = torch.empty_like(input1)
    else:
        if out.shape != input1.shape:
            raise ValueError("out tensor must have the same shape as input1")
    
    # Flatten tensors for processing
    input1_flat = input1.flatten()
    input2_flat = input2.flatten()
    out_flat = out.flatten()
    
    # Get dimensions
    n_elements = input1_flat.numel()
    channels = input1_flat.shape[-1]  # Assuming last dimension is channels
    
    # Validate group normalization parameters
    if channels % num_groups != 0:
        raise ValueError("Number of channels must be divisible by num_groups")
    
    # Set up kernel launch parameters
    BLOCK_SIZE = 256
    grid = (triton.cdiv(n_elements, BLOCK_SIZE),)
    
    # Create stride tensors
    x_stride = input1.stride(-1) if input1.dim() > 0 else 1
    y_stride = input2.stride(-1) if input2.dim() > 0 else 1
    weight_stride = weight.stride(-1) if weight.dim() > 0 else 1
    bias_stride = bias.stride(-1) if bias.dim() > 0 else 1
    out_stride = out.stride(-1) if out.dim() > 0 else 1
    
    # Launch kernel
    _fused_add_mul_groupnorm_kernel[grid](
        input1_flat, input2_flat, weight, bias, out_flat,
        n_elements, num_groups, channels, channels // num_groups, eps,
        x_stride, y_stride, weight_stride, bias_stride, out_stride,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    return out
