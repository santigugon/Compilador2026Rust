import torch
import triton
import triton.language as tl

@triton.jit
def combined_activation_kernel(
    input_ptr, weight1_ptr, weight2_ptr, bias_ptr, output_ptr,
    input_row_stride, weight1_row_stride, weight2_row_stride, bias_row_stride,
    output_row_stride,
    N, D_in, D_out,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start_m = pid * BLOCK_SIZE_M
    block_start_n = pid * BLOCK_SIZE_N
    
    # Load input and weight1
    input_block = tl.load(input_ptr + block_start_m * input_row_stride + tl.arange(0, BLOCK_SIZE_M)[:, None] * input_row_stride + tl.arange(0, D_in)[None, :] * 1)
    weight1_block = tl.load(weight1_ptr + tl.arange(0, D_in)[:, None] * weight1_row_stride + tl.arange(0, D_out)[None, :] * 1)
    
    # Matrix multiplication
    intermediate = tl.dot(input_block, weight1_block)
    
    # Apply sigmoid and tanh
    intermediate = tl.sigmoid(intermediate)
    intermediate = tl.tanh(intermediate)
    
    # Element-wise multiplication with weight2
    weight2_block = tl.load(weight2_ptr + tl.arange(0, D_out)[None, :] * weight2_row_stride)
    intermediate = intermediate * weight2_block
    
    # Add bias
    bias_block = tl.load(bias_ptr + tl.arange(0, D_out)[None, :] * bias_row_stride)
    intermediate = intermediate + bias_block
    
    # Store result
    output_block = intermediate
    tl.store(output_ptr + block_start_m * output_row_stride + tl.arange(0, BLOCK_SIZE_M)[:, None] * output_row_stride + tl.arange(0, D_out)[None, :] * 1, output_block)

def combined_activation(input, weight1, weight2, bias, *, out=None):
    if out is None:
        out = torch.empty(input.shape[:-1] + (weight1.shape[1],), dtype=input.dtype, device=input.device)
    
    # Ensure input and weight1 are compatible for matrix multiplication
    assert input.shape[-1] == weight1.shape[0], "Input and weight1 dimensions are not compatible"
    
    # Ensure weight2 and bias are broadcastable to output shape
    assert weight2.shape[-1] == weight1.shape[1], "Weight2 and weight1 dimensions are not compatible"
    assert bias.shape[-1] == weight1.shape[1], "Bias and weight1 dimensions are not compatible"
    
    # Flatten batch dimensions
    batch_shape = input.shape[:-1]
    input_flat = input.view(-1, input.shape[-1])
    out_flat = out.view(-1, out.shape[-1])
    
    # Get dimensions
    N = input_flat.shape[0]
    D_in = input_flat.shape[1]
    D_out = weight1.shape[1]
    
    # Launch kernel
    grid = (triton.cdiv(N, 32) * triton.cdiv(D_out, 32),)
    combined_activation_kernel[grid](
        input_flat, weight1, weight2, bias, out_flat,
        input_flat.stride(0), weight1.stride(0), weight2.stride(0), bias.stride(0),
        out_flat.stride(0),
        N, D_in, D_out,
        BLOCK_SIZE_M=32,
        BLOCK_SIZE_N=32,
        BLOCK_SIZE_K=32,
    )
    
    return out
