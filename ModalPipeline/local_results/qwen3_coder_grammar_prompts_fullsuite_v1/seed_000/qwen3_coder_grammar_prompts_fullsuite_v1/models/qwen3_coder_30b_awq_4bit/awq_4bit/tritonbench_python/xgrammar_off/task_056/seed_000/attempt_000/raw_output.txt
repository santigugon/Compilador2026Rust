import torch
import triton
import triton.language as tl

@triton.jit
def relu_and_pool2d_kernel(
    input_ptr,
    output_ptr,
    indices_ptr,
    input_shape,
    output_shape,
    kernel_size,
    output_ratio,
    return_indices,
    BLOCK_SIZE_H=16,
    BLOCK_SIZE_W=16,
):
    # Get block indices
    block_idx_h = tl.program_id(0)
    block_idx_w = tl.program_id(1)
    
    # Calculate output indices
    out_h = block_idx_h * BLOCK_SIZE_H
    out_w = block_idx_w * BLOCK_SIZE_W
    
    # Load input data
    input_ptr = input_ptr + out_h * input_shape[2] + out_w
    
    # Apply ReLU and perform fractional max pooling
    for i in range(BLOCK_SIZE_H):
        for j in range(BLOCK_SIZE_W):
            if out_h + i < output_shape[0] and out_w + j < output_shape[1]:
                # Apply ReLU
                input_val = tl.load(input_ptr + i * input_shape[2] + j)
                relu_val = tl.maximum(input_val, 0.0)
                
                # Fractional max pooling
                max_val = relu_val
                max_idx_h = 0
                max_idx_w = 0
                
                # Simple fractional max pooling logic
                for kh in range(kernel_size):
                    for kw in range(kernel_size):
                        h = out_h + i + kh
                        w = out_w + j + kw
                        if h < input_shape[0] and w < input_shape[1]:
                            val = tl.load(input_ptr + h * input_shape[2] + w)
                            relu_val = tl.maximum(val, 0.0)
                            if relu_val > max_val:
                                max_val = relu_val
                                max_idx_h = h
                                max_idx_w = w
                
                # Store output
                output_ptr = output_ptr + out_h * output_shape[2] + out_w
                tl.store(output_ptr + i * output_shape[2] + j, max_val)
                
                # Store indices if needed
                if return_indices:
                    indices_ptr = indices_ptr + out_h * output_shape[2] + out_w
                    tl.store(indices_ptr + i * output_shape[2] + j, max_idx_h * output_shape[2] + max_idx_w)

def fused_fractional_max_pool2d_with_relu(input: torch.Tensor, kernel_size, output_size=None, output_ratio=None, return_indices=False) -> torch.Tensor:
    # Validate input
    if input.dim() != 4:
        raise ValueError("Input tensor must be 4-dimensional (N, C, H, W)")
    
    # Get input dimensions
    batch_size, channels, height, width = input.shape
    
    # Determine output size
    if output_size is not None:
        output_h, output_w = output_size
    elif output_ratio is not None:
        output_h = int(height * output_ratio[0])
        output_w = int(width * output_ratio[1])
    else:
        # Default to 1x1 output
        output_h, output_w = 1, 1
    
    # Create output tensor
    output = torch.empty(batch_size, channels, output_h, output_w, device=input.device, dtype=input.dtype)
    
    # Create indices tensor if needed
    indices = None
    if return_indices:
        indices = torch.empty(batch_size, channels, output_h, output_w, device=input.device, dtype=torch.long)
    
    # Launch kernel
    if return_indices:
        relu_and_pool2d_kernel[(output_h, output_w)](
            input,
            output,
            indices,
            input.shape,
            output.shape,
            kernel_size,
            output_ratio,
            return_indices
        )
    else:
        relu_and_pool2d_kernel[(output_h, output_w)](
            input,
            output,
            None,
            input.shape,
            output.shape,
            kernel_size,
            output_ratio,
            return_indices
        )
    
    # Return appropriate output
    if return_indices:
        return output, indices
    else:
        return output
