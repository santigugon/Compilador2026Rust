import torch
import triton
import triton.language as tl

@triton.jit
def mean_kernel(X, Y, stride_x, stride_y, N, M, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < M
    x = tl.load(X + offsets * stride_x, mask=mask)
    mean_val = tl.sum(x) / N
    tl.store(Y + pid * stride_y, mean_val)

def mean_triton(input, dim, keepdim=False, dtype=None, out=None):
    if isinstance(dim, int):
        dim = (dim,)
    
    if dtype is not None:
        input = input.to(dtype)
    
    input_flat = input
    for d in sorted(dim, reverse=True):
        input_flat = input_flat.sum(dim=d, keepdim=True)
    
    if keepdim:
        output = input_flat
    else:
        output = input_flat.squeeze(dim)
    
    return output
