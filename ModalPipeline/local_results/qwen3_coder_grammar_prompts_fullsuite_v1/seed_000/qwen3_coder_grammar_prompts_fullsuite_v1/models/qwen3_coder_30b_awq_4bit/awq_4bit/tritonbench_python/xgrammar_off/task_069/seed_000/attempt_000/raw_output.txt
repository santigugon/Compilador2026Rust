import torch
import triton
import triton.language as tl

@triton.jit
def fused_layer_norm_relu_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_row_stride, weight_row_stride, weight_col_stride,
    output_row_stride, output_col_stride,
    n_rows, n_cols, out_features,
    eps,
    BLOCK_SIZE: tl.constexpr,
    GROUP_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    # Load input row
    input_row = tl.load(input_ptr + row_idx * input_row_stride + tl.arange(0, BLOCK_SIZE), mask=tl.arange(0, BLOCK_SIZE) < n_cols)
    
    # Linear transformation
    output_row = tl.zeros((out_features,), dtype=tl.float32)
    for i in range(0, n_cols, BLOCK_SIZE):
        col_mask = (tl.arange(0, BLOCK_SIZE) + i) < n_cols
        input_vals = tl.load(input_ptr + row_idx * input_row_stride + i + tl.arange(0, BLOCK_SIZE), mask=col_mask)
        for j in range(out_features):
            weight_vals = tl.load(weight_ptr + j * weight_row_stride + i + tl.arange(0, BLOCK_SIZE), mask=col_mask)
            output_row = tl.where(i == 0, output_row + input_vals * weight_vals, output_row + input_vals * weight_vals)
    
    # Add bias
    if bias_ptr is not None:
        bias_vals = tl.load(bias_ptr + tl.arange(0, out_features))
        output_row = output_row + bias_vals
    
    # Apply ReLU
    output_row = tl.where(output_row > 0, output_row, 0.0)
    
    # Layer normalization
    mean = tl.sum(output_row) / out_features
    var = tl.sum((output_row - mean) * (output_row - mean)) / out_features
    std = tl.sqrt(var + eps)
    output_row = (output_row - mean) / std
    
    # Store result
    tl.store(output_ptr + row_idx * output_row_stride + tl.arange(0, out_features), output_row)

def fused_layer_norm_relu_linear(input, weight, bias=None, normalized_shape=None, eps=1e-5, elementwise_affine=True):
    if normalized_shape is None:
        normalized_shape = weight.shape[0]
    if isinstance(normalized_shape, int):
        normalized_shape = [normalized_shape]
    
    # Validate input shapes
    assert input.dim() >= 2
    assert weight.dim() == 2
    assert weight.shape[1] == input.shape[-1]
    assert weight.shape[0] == (normalized_shape[0] if len(normalized_shape) == 1 else normalized_shape[-1])
    
    # Prepare output shape
    output_shape = list(input.shape[:-1]) + [weight.shape[0]]
    
    # Create output tensor
    output = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    
    # Prepare strides
    n_rows = input.numel() // input.shape[-1]
    n_cols = input.shape[-1]
    out_features = weight.shape[0]
    
    # Launch kernel
    BLOCK_SIZE = 256
    GROUP_SIZE = 8
    
    grid = lambda meta: (triton.cdiv(n_rows, meta['BLOCK_SIZE']),)
    
    # Create a kernel that handles the fused operations
    fused_layer_norm_relu_linear_kernel[
        grid,
        (BLOCK_SIZE, GROUP_SIZE)
    ](
        input_ptr=input.data_ptr(),
        weight_ptr=weight.data_ptr(),
        bias_ptr=bias.data_ptr() if bias is not None else None,
        output_ptr=output.data_ptr(),
        input_row_stride=n_cols,
        weight_row_stride=weight.shape[1],
        weight_col_stride=1,
        output_row_stride=out_features,
        output_col_stride=1,
        n_rows=n_rows,
        n_cols=n_cols,
        out_features=out_features,
        eps=eps,
        BLOCK_SIZE=BLOCK_SIZE,
        GROUP_SIZE=GROUP_SIZE
    )
    
    return output
