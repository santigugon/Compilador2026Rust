import torch
import triton
import triton.language as tl

@triton.jit
def rms_norm_kernel(X_ptr, Y_ptr, weight_ptr, eps, N, P, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < N
    x = tl.load(X_ptr + offsets, mask=mask)
    y = tl.load(Y_ptr + offsets, mask=mask)
    weight = tl.load(weight_ptr + offsets, mask=mask)
    mean = tl.sum(x * x, axis=0) / P
    inv_rms = tl.rsqrt(mean + eps)
    output = y * inv_rms * weight
    tl.store(Y_ptr + offsets, output, mask=mask)

@triton.jit
def gelu_kernel(X_ptr, Y_ptr, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    x = tl.load(X_ptr + offsets)
    y = 0.5 * x * (1 + tl.tanh(0.7978845608 * (x + 0.044715 * x * x * x)))
    tl.store(Y_ptr + offsets, y)

@triton.jit
def dropout_kernel(X_ptr, Y_ptr, mask_ptr, dropout_p, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    x = tl.load(X_ptr + offsets)
    mask = tl.load(mask_ptr + offsets)
    y = x * mask / (1.0 - dropout_p)
    tl.store(Y_ptr + offsets, y)

@triton.jit
def subtract_kernel(X_ptr, Y_ptr, Z_ptr, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    x = tl.load(X_ptr + offsets)
    y = tl.load(Y_ptr + offsets)
    z = tl.load(Z_ptr + offsets)
    result = x - z
    tl.store(Y_ptr + offsets, result)

def fused_bmm_rmsnorm_gelu_dropout_sub(input1, input2, other, normalized_shape, dropout_p=0.5, training=True, approximate='none', eps=1e-5, *, out=None):
    B, N, M = input1.shape
    _, _, P = input2.shape
    if out is None:
        out = torch.empty(B, N, P, device=input1.device, dtype=input1.dtype)
    
    # Batch matrix multiplication
    output = torch.bmm(input1, input2)
    
    # RMS normalization
    weight = torch.ones(normalized_shape, device=input1.device, dtype=input1.dtype)
    output = output.view(-1, P)
    weight = weight.expand_as(output)
    output = output.contiguous()
    weight = weight.contiguous()
    
    # Apply RMS normalization using Triton kernel
    BLOCK_SIZE = 1024
    num_blocks = (N * P + BLOCK_SIZE - 1) // BLOCK_SIZE
    rms_norm_kernel[(num_blocks,)](output, output, weight, eps, N * P, P, BLOCK_SIZE=BLOCK_SIZE)
    
    # GELU activation
    output = output.view(B, N, P)
    output = output.contiguous()
    gelu_kernel[(num_blocks,)](output, output, BLOCK_SIZE=BLOCK_SIZE)
    
    # Dropout
    if training:
        dropout_mask = torch.rand(B, N, P, device=input1.device) > dropout_p
        dropout_mask = dropout_mask.to(input1.dtype)
        dropout_mask = dropout_mask.contiguous()
        dropout_kernel[(num_blocks,)](output, output, dropout_mask, dropout_p, BLOCK_SIZE=BLOCK_SIZE)
    
    # Subtraction
    other = other.expand_as(output)
    other = other.contiguous()
    subtract_kernel[(num_blocks,)](output, output, other, BLOCK_SIZE=BLOCK_SIZE)
    
    return output
