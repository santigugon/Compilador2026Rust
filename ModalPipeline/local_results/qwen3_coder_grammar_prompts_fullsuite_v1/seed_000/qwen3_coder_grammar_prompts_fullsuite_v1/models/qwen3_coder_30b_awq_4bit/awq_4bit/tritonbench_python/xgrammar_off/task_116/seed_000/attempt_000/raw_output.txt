import triton.language as tl

@triton.jit
def sum_kernel(
    input_ptr,
    output_ptr,
    input_row_stride,
    output_row_stride,
    n_rows,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
    keepdim: tl.constexpr,
    reduce_dim: tl.constexpr
):
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
