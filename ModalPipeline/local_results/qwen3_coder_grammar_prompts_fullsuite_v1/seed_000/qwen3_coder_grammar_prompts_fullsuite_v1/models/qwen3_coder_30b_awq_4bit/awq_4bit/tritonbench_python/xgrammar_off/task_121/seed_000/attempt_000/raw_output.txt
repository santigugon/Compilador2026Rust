import torch
import triton
import triton.language as tl

@triton.jit
def gelu_kernel(X, Y, N, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < N
    x = tl.load(X + offsets, mask=mask)
    # GELU approximation using tanh
    y = 0.5 * x * (1 + tl.tanh(0.7978845608 * (x + 0.044715 * x * x * x)))
    tl.store(Y + offsets, y, mask=mask)

@triton.jit
def gelu_exact_kernel(X, Y, N, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < N
    x = tl.load(X + offsets, mask=mask)
    # Exact GELU computation
    y = 0.5 * x * (1 + tl.erf(x / tl.sqrt(2.0)))
    tl.store(Y + offsets, y, mask=mask)

@triton.jit
def min_kernel(X, Y, N, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < N
    x = tl.load(X + offsets, mask=mask)
    # Find minimum value
    min_val = tl.min(x)
    tl.store(Y + pid, min_val, mask=pid < 1)

def min_gelu(input, dim=None, keepdim=False, approximate='none', out=None):
    if out is not None:
        raise NotImplementedError("out parameter is not supported")
    
    input = input.contiguous()
    if approximate == 'none':
        # Use exact GELU
        kernel = gelu_exact_kernel
    else:
        # Use tanh approximation
        kernel = gelu_kernel
    
    # Compute GELU
    output = torch.empty_like(input)
    N = input.numel()
    BLOCK_SIZE = 1024
    grid = (triton.cdiv(N, BLOCK_SIZE),)
    kernel[grid](input, output, N, BLOCK_SIZE=BLOCK_SIZE)
    
    # Compute minimum
    if dim is None:
        # Reduce all elements
        result = torch.min(output)
        if keepdim:
            result = result.unsqueeze(0)
    else:
        # Reduce along specified dimension
        result = torch.min(output, dim=dim, keepdim=keepdim)[0]
    
    return result
