import torch
import triton
import triton.language as tl

@triton.jit
def pairwise_distance_kernel(x1_ptr, x2_ptr, output_ptr, x1_size, x2_size, p, eps, BLOCK_SIZE):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < x1_size
    x1_data = tl.load(x1_ptr + offsets, mask=mask)
    x2_data = tl.load(x2_ptr + offsets, mask=mask)
    diff = x1_data - x2_data
    diff = diff * diff
    diff = tl.sum(diff, axis=0)
    diff = tl.sqrt(diff + eps)
    tl.store(output_ptr + pid, diff)

@triton.jit
def normalize_kernel(input_ptr, output_ptr, input_size, p, eps, BLOCK_SIZE):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < input_size
    input_data = tl.load(input_ptr + offsets, mask=mask)
    input_data = tl.abs(input_data)
    input_data = tl.pow(input_data, p)
    input_data = tl.sum(input_data, axis=0)
    input_data = tl.pow(input_data, 1.0 / p)
    input_data = input_data + eps
    input_data = tl.load(input_ptr + offsets, mask=mask) / input_data
    tl.store(output_ptr + pid, input_data)

def normalize_pairwise_distance(x1, x2, p_distance=2.0, eps_distance=1e-6, keepdim=False, p_norm=2, dim_norm=1, eps_norm=1e-12):
    x1 = x1.contiguous()
    x2 = x2.contiguous()
    output = torch.empty(x1.shape[0], x2.shape[0], dtype=torch.float32, device=x1.device)
    
    # Compute pairwise distances
    BLOCK_SIZE = 1024
    num_blocks = (x1.shape[0] * x2.shape[0] + BLOCK_SIZE - 1) // BLOCK_SIZE
    
    for i in range(x1.shape[0]):
        for j in range(x2.shape[0]):
            x1_ptr = x1[i].contiguous()
            x2_ptr = x2[j].contiguous()
            output_ptr = output[i, j].contiguous()
            pairwise_distance_kernel[(num_blocks,)](x1_ptr, x2_ptr, output_ptr, x1.shape[0], x2.shape[0], p_distance, eps_distance, BLOCK_SIZE=BLOCK_SIZE)
    
    # Normalize along specified dimension
    if dim_norm == 1:
        output = output / (torch.norm(output, p=p_norm, dim=dim_norm, keepdim=keepdim) + eps_norm)
    else:
        output = output / (torch.norm(output, p=p_norm, dim=dim_norm, keepdim=keepdim) + eps_norm)
    
    return output
