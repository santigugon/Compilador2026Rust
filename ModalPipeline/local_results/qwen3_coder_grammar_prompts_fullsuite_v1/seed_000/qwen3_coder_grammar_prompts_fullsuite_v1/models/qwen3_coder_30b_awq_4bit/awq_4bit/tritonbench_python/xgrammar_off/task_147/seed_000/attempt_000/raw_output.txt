import torch
import triton
import triton.language as tl

@triton.jit
def normalize_kernel(x_ptr, output_ptr, n, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask)
    norm = tl.sqrt(tl.sum(x * x) + 1e-12)
    normalized = x / norm
    tl.store(output_ptr + offsets, normalized, mask=mask)

@triton.jit
def pairwise_distance_kernel(x1_ptr, x2_ptr, output_ptr, n1, n2, p_norm, eps_distance, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n1 * n2
    x1_offsets = offsets // n2
    x2_offsets = offsets % n2
    x1_val = tl.load(x1_ptr + x1_offsets, mask=x1_offsets < n1)
    x2_val = tl.load(x2_ptr + x2_offsets, mask=x2_offsets < n2)
    diff = x1_val - x2_val
    if p_norm == 2.0:
        distance = tl.sqrt(tl.sum(diff * diff) + eps_distance)
    else:
        distance = tl.pow(tl.sum(tl.abs(diff) ** p_norm), 1.0 / p_norm)
    tl.store(output_ptr + offsets, distance, mask=mask)

def fused_pairwise_distance_normalize(x1: torch.Tensor, x2: torch.Tensor, p_norm: float = 2.0, eps_norm: float = 1e-12, eps_distance: float = 1e-6, keepdim: bool = False) -> torch.Tensor:
    x1 = x1.float()
    x2 = x2.float()
    
    # Normalize x1 and x2
    x1_normalized = torch.empty_like(x1)
    x2_normalized = torch.empty_like(x2)
    
    # Normalize along the last dimension
    n = x1.shape[-1]
    BLOCK_SIZE = 1024
    grid = (triton.cdiv(n, BLOCK_SIZE),)
    
    normalize_kernel[grid](x1, x1_normalized, n, BLOCK_SIZE=BLOCK_SIZE)
    normalize_kernel[grid](x2, x2_normalized, n, BLOCK_SIZE=BLOCK_SIZE)
    
    # Compute pairwise distances
    n1 = x1_normalized.shape[-1]
    n2 = x2_normalized.shape[-1]
    output = torch.empty(x1_normalized.shape[:-1] + x2_normalized.shape[:-1], dtype=torch.float32)
    
    # Reshape for computation
    x1_flat = x1_normalized.view(-1, n1)
    x2_flat = x2_normalized.view(-1, n2)
    output_flat = output.view(-1, n2)
    
    # Compute pairwise distances
    n1_total = x1_flat.shape[0]
    n2_total = x2_flat.shape[0]
    total_elements = n1_total * n2_total
    grid = (triton.cdiv(total_elements, BLOCK_SIZE),)
    
    pairwise_distance_kernel[grid](x1_flat, x2_flat, output_flat, n1_total, n2_total, p_norm, eps_distance, BLOCK_SIZE=BLOCK_SIZE)
    
    if keepdim:
        output = output.unsqueeze(-1)
    
    return output
