import torch
import triton
import triton.language as tl

@triton.jit
def _qr_decomposition_kernel(A, R, Q, n, block_size=16):
    # Shared memory for intermediate computations
    shared_A = tl.shared_memory(shape=(block_size, block_size), dtype=tl.float32)
    shared_R = tl.shared_memory(shape=(block_size, block_size), dtype=tl.float32)
    
    # Thread and block indices
    tx = tl.program_id(0)
    ty = tl.program_id(1)
    
    # Load data into shared memory
    if tx < n and ty < n:
        shared_A[tx, ty] = A[tx, ty]
    
    # Synchronize to ensure all threads have loaded data
    tl.sync()
    
    # Initialize R and Q matrices
    if tx < n and ty < n:
        if tx == ty:
            R[tx, ty] = shared_A[tx, ty]
        else:
            R[tx, ty] = 0.0
            Q[tx, ty] = 0.0
    
    # Synchronize to ensure initialization is complete
    tl.sync()
    
    # Perform QR decomposition using Givens rotations
    for k in range(min(n, block_size)):
        # Compute Givens rotation
        if tx == k and ty == k:
            # Compute norm of k-th column
            norm = 0.0
            for i in range(k, n):
                norm += shared_A[i, k] * shared_A[i, k]
            norm = tl.sqrt(norm)
            
            # Update R and Q
            if norm > 0.0:
                R[k, k] = norm
                Q[k, k] = 1.0
                for i in range(k + 1, n):
                    Q[i, k] = shared_A[i, k] / norm
                    R[i, k] = 0.0
            else:
                R[k, k] = 0.0
                Q[k, k] = 0.0
                for i in range(k + 1, n):
                    Q[i, k] = 0.0
                    R[i, k] = 0.0
        
        # Synchronize to ensure Givens rotation is computed
        tl.sync()
        
        # Apply Givens rotation to remaining columns
        for j in range(k + 1, n):
            if tx == k and ty == j:
                # Compute cosine and sine
                cos = R[k, k] / (R[k, k] + R[k, j])
                sin = R[k, j] / (R[k, k] + R[k, j])
                
                # Update R matrix
                R[k, j] = cos * R[k, j] + sin * R[k, k]
                R[k, k] = cos * R[k, k] - sin * R[k, j]
                
                # Update Q matrix
                Q[k, j] = cos * Q[k, j] + sin * Q[k, k]
                Q[k, k] = cos * Q[k, k] - sin * Q[k, j]
        
        # Synchronize to ensure updates are complete
        tl.sync()

@triton.jit
def _determinant_kernel(R, det, n):
    # Compute determinant as product of diagonal elements
    det_val = 1.0
    for i in range(n):
        det_val *= R[i, i]
    
    # Store result
    det[0] = det_val

def determinant_via_qr(A, *, mode='reduced', out=None):
    # Validate input
    if not isinstance(A, torch.Tensor):
        raise TypeError("Input must be a torch.Tensor")
    
    if A.dim() != 2:
        raise ValueError("Input must be a 2D tensor")
    
    if A.size(0) != A.size(1):
        raise ValueError("Input must be a square matrix")
    
    n = A.size(0)
    
    # Ensure input is on GPU and float32
    if A.device.type != 'cuda':
        A = A.cuda()
    
    if A.dtype != torch.float32:
        A = A.float()
    
    # Allocate output tensor
    if out is None:
        out = torch.empty(1, dtype=torch.float32, device=A.device)
    else:
        if out.shape != (1,) or out.dtype != torch.float32:
            raise ValueError("Output tensor must be of shape (1,) and dtype torch.float32")
    
    # Allocate memory for R and Q matrices
    R = torch.empty_like(A)
    Q = torch.empty_like(A)
    
    # Launch QR decomposition kernel
    grid = (triton.cdiv(n, 16), triton.cdiv(n, 16))
    _qr_decomposition_kernel[grid](A, R, Q, n)
    
    # Launch determinant kernel
    _determinant_kernel[1](R, out, n)
    
    return out
