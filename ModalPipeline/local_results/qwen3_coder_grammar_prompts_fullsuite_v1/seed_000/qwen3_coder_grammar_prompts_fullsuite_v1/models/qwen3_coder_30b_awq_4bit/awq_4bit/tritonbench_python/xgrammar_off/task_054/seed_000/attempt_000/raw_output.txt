import torch
import triton
import triton.language as tl
from typing import Optional, Union, Tuple

@triton.jit
def gelu_kernel(
    input_ptr, 
    weight_ptr, 
    bias_ptr, 
    output_ptr,
    input_shape,
    weight_shape,
    bias_shape,
    output_shape,
    stride_h,
    stride_w,
    padding_h,
    padding_w,
    dilation_h,
    dilation_w,
    groups,
    approximate,
    BLOCK_SIZE_H,
    BLOCK_SIZE_W,
    BLOCK_SIZE_C,
    BLOCK_SIZE_K,
    num_warps=4
):
    # Get the block index
    block_idx = tl.program_id(0)
    # Calculate the output indices
    out_h = block_idx // output_shape[3]
    out_w = block_idx % output_shape[3]
    
    # Load input and weight
    input_block = tl.load(input_ptr + out_h * stride_h + out_w * stride_w)
    weight_block = tl.load(weight_ptr)
    
    # Perform convolution
    conv_result = tl.sum(input_block * weight_block, axis=0)
    
    # Apply bias if present
    if bias_ptr is not None:
        bias_block = tl.load(bias_ptr)
        conv_result += bias_block
    
    # Apply GELU activation
    if approximate == 'none':
        gelu_result = 0.5 * conv_result * (1 + tl.erf(conv_result / tl.sqrt(2.0)))
    else:
        gelu_result = conv_result * 0.5 * (1 + tl.tanh(0.7978845608 * (conv_result + 0.044715 * conv_result * conv_result * conv_result)))
    
    # Store the result
    tl.store(output_ptr + out_h * output_shape[3] + out_w, gelu_result)

def gelu_conv2d(
    input: torch.Tensor, 
    weight: torch.Tensor, 
    bias: Optional[torch.Tensor] = None, 
    stride: Union[int, Tuple[int, int]] = 1, 
    padding: Union[int, Tuple[int, int], str] = 0, 
    dilation: Union[int, Tuple[int, int]] = 1, 
    groups: int = 1, 
    approximate: str = 'none', 
    out: Optional[torch.Tensor] = None
) -> torch.Tensor:
    # Convert stride, padding, and dilation to tuples
    if isinstance(stride, int):
        stride = (stride, stride)
    if isinstance(padding, int):
        padding = (padding, padding)
    if isinstance(dilation, int):
        dilation = (dilation, dilation)
    
    # Get input dimensions
    batch_size, in_channels, iH, iW = input.shape
    out_channels, _, kH, kW = weight.shape
    
    # Calculate output dimensions
    oH = (iH + 2 * padding[0] - (kH - 1) * dilation[0] - 1) // stride[0] + 1
    oW = (iW + 2 * padding[1] - (kW - 1) * dilation[1] - 1) // stride[1] + 1
    
    # Initialize output tensor
    if out is None:
        out = torch.empty(batch_size, out_channels, oH, oW, dtype=input.dtype, device=input.device)
    
    # Prepare kernel launch parameters
    BLOCK_SIZE_H = 16
    BLOCK_SIZE_W = 16
    BLOCK_SIZE_C = 32
    BLOCK_SIZE_K = 32
    
    # Launch kernel
    grid = (oH * oW, 1, 1)
    gelu_kernel[grid](
        input,
        weight,
        bias,
        out,
        input.shape,
        weight.shape,
        bias.shape if bias is not None else None,
        out.shape,
        stride[0],
        stride[1],
        padding[0],
        padding[1],
        dilation[0],
        dilation[1],
        groups,
        approximate,
        BLOCK_SIZE_H,
        BLOCK_SIZE_W,
        BLOCK_SIZE_C,
        BLOCK_SIZE_K
    )
    
    return out
