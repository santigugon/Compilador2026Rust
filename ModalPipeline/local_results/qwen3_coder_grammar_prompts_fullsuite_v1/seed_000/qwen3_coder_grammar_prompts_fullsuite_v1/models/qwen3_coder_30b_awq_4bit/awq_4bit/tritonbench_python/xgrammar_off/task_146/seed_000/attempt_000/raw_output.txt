import torch
import triton
import triton.language as tl

@triton.jit
def elu_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_size, weight_size, bias_size,
    alpha, inplace,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < input_size
    
    # Load input
    input = tl.load(input_ptr + offsets, mask=mask)
    
    # Linear transformation
    linear_output = tl.dot(input, weight_ptr)
    
    # Add bias if present
    if bias_ptr is not None:
        bias = tl.load(bias_ptr + tl.arange(0, weight_size), mask=tl.arange(0, weight_size) < bias_size)
        linear_output = linear_output + bias
    
    # Apply ELU
    elu_output = tl.where(linear_output > 0, linear_output, alpha * (tl.exp(linear_output) - 1))
    
    # Store result
    tl.store(output_ptr + offsets, elu_output, mask=mask)

def elu_linear(input, weight, bias=None, alpha=1.0, inplace=False):
    if inplace:
        raise NotImplementedError("In-place ELU linear is not implemented in Triton")
    
    # Ensure inputs are contiguous
    input = input.contiguous()
    weight = weight.contiguous()
    
    # Compute output size
    output_size = input.size(0)
    weight_size = weight.size(1)
    
    # Create output tensor
    output = torch.empty(output_size, weight_size, device=input.device, dtype=input.dtype)
    
    # Prepare pointers
    input_ptr = input.data_ptr()
    weight_ptr = weight.data_ptr()
    bias_ptr = bias.data_ptr() if bias is not None else None
    output_ptr = output.data_ptr()
    
    # Launch kernel
    BLOCK_SIZE = 1024
    num_blocks = (output_size + BLOCK_SIZE - 1) // BLOCK_SIZE
    
    elu_linear_kernel[
        num_blocks,
        1,
        (input_ptr, weight_ptr, bias_ptr, output_ptr, input.numel(), weight.size(1), bias.numel() if bias is not None else 0, alpha, inplace, BLOCK_SIZE)
    ]()
    
    return output
