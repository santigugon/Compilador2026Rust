import torch
import triton
import triton.language as tl

@triton.jit
def bessel_j1_kernel(x_ptr, output_ptr, n, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # Compute Bessel function of the first kind of order 1
    # Using approximation: J1(x) ≈ (x/2) * (1 - x^2/8 + x^4/192 - x^6/9216 + ...)
    # For small x, use series expansion
    # For large x, use asymptotic expansion
    
    # Simplified implementation for demonstration
    # In practice, this would use more sophisticated numerical methods
    x_sq = x * x
    x_sq_2 = x_sq * 0.25
    
    # Series expansion for small x
    term = x_sq_2
    result = x_sq_2
    
    # Add more terms for better accuracy
    term = term * (-x_sq / 8)
    result = result + term
    
    term = term * (-x_sq / 192)
    result = result + term
    
    term = term * (-x_sq / 9216)
    result = result + term
    
    # Handle special cases
    # For x = 0, J1(0) = 0
    # For large x, use asymptotic form
    mask_zero = (x == 0.0)
    result = tl.where(mask_zero, 0.0, result)
    
    tl.store(output_ptr + offsets, result, mask=mask)

def bessel_j1(input, *, out=None):
    if out is None:
        out = torch.empty_like(input, dtype=torch.float32, device=input.device)
    
    # Ensure input is float32
    if input.dtype != torch.float32:
        input = input.float()
    
    n = input.numel()
    BLOCK_SIZE = 256
    grid = (triton.cdiv(n, BLOCK_SIZE),)
    
    bessel_j1_kernel[grid](
        input.data_ptr(),
        out.data_ptr(),
        n,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    return out
