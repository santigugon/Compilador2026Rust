import torch
import triton
import triton.language as tl

@triton.jit
def fused_hardsigmoid_batch_norm_kernel(
    x_ptr, 
    running_mean_ptr, 
    running_var_ptr, 
    weight_ptr, 
    bias_ptr,
    output_ptr,
    N,
    C,
    eps,
    momentum,
    training,
    BLOCK_SIZE=1024
):
    # Get the block index
    block_idx = tl.program_id(0)
    # Calculate the start index for this block
    start_idx = block_idx * BLOCK_SIZE
    # Calculate the end index for this block
    end_idx = min(start_idx + BLOCK_SIZE, N)
    
    # Load data for this block
    x_block = tl.load(x_ptr + start_idx, mask=start_idx < N)
    
    # Normalize the data
    mean = tl.load(running_mean_ptr)
    var = tl.load(running_var_ptr)
    x_normalized = (x_block - mean) / tl.sqrt(var + eps)
    
    # Apply weight and bias if provided
    if weight_ptr is not None and bias_ptr is not None:
        weight = tl.load(weight_ptr)
        bias = tl.load(bias_ptr)
        x_normalized = x_normalized * weight + bias
    
    # Apply Hardsigmoid activation
    output = tl.where(x_normalized >= 0, tl.where(x_normalized <= 1, x_normalized, 1.0), 0.0)
    
    # Store the result
    tl.store(output_ptr + start_idx, output, mask=start_idx < N)

def fused_hardsigmoid_batch_norm(
    x: torch.Tensor, 
    running_mean: torch.Tensor, 
    running_var: torch.Tensor, 
    weight: torch.Tensor = None, 
    bias: torch.Tensor = None, 
    training: bool = False, 
    momentum: float = 0.1, 
    eps: float = 1e-5, 
    inplace: bool = False
) -> torch.Tensor:
    # Ensure input tensor is contiguous
    x = x.contiguous()
    
    # Get the total number of elements and channels
    N = x.numel()
    C = x.shape[1] if len(x.shape) > 1 else 1
    
    # Create output tensor
    if inplace:
        output = x
    else:
        output = torch.empty_like(x)
    
    # Prepare kernel arguments
    grid = (triton.cdiv(N, 1024),)
    
    # Launch kernel
    fused_hardsigmoid_batch_norm_kernel[grid](
        x_ptr=x.data_ptr(),
        running_mean_ptr=running_mean.data_ptr(),
        running_var_ptr=running_var.data_ptr(),
        weight_ptr=weight.data_ptr() if weight is not None else None,
        bias_ptr=bias.data_ptr() if bias is not None else None,
        output_ptr=output.data_ptr(),
        N=N,
        C=C,
        eps=eps,
        momentum=momentum,
        training=training
    )
    
    return output
