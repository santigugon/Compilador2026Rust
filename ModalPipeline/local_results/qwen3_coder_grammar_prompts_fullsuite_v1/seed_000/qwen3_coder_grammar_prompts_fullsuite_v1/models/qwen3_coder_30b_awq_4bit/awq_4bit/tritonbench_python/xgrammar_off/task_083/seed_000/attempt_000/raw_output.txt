import torch
import triton
import triton.language as tl

@triton.jit
def _fused_bmm_dropout_gelu_kernel(
    input1_ptr, input2_ptr, output_ptr, 
    dropout_mask_ptr,
    B, N, M, P, 
    p, 
    training,
    approximate,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
    BLOCK_SIZE_O: tl.constexpr
):
    # Get block indices
    block_idx = tl.program_id(0)
    block_idy = tl.program_id(1)
    block_idz = tl.program_id(2)
    
    # Compute block start positions
    start_m = block_idx * BLOCK_SIZE_M
    start_n = block_idy * BLOCK_SIZE_N
    start_k = block_idz * BLOCK_SIZE_K
    
    # Load input tensors
    input1 = tl.load(input1_ptr + start_m * M + start_k)
    input2 = tl.load(input2_ptr + start_k * P + start_n)
    
    # Compute batch matrix multiplication
    acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, M, BLOCK_SIZE_K):
        a = tl.load(input1_ptr + start_m * M + k)
        b = tl.load(input2_ptr + k * P + start_n)
        acc += tl.dot(a, b)
    
    # Apply dropout if training
    if training:
        # Generate dropout mask
        mask = tl.rand() > p
        acc = tl.where(mask, acc / (1.0 - p), 0.0)
    
    # Apply GELU activation
    if approximate == 'tanh':
        # Approximate GELU using tanh
        acc = 0.5 * acc * (1 + tl.tanh(0.7978845608 * (acc + 0.044715 * acc * acc * acc)))
    else:
        # Exact GELU
        acc = 0.5 * acc * (1 + tl.erf(acc / tl.sqrt(2.0)))
    
    # Store result
    tl.store(output_ptr + start_m * P + start_n, acc)

def fused_bmm_dropout_gelu(input1, input2, p=0.5, training=True, inplace=False, approximate='none', *, out=None):
    # Validate inputs
    assert input1.dim() == 3 and input2.dim() == 3
    assert input1.size(0) == input2.size(0)  # Batch size
    assert input1.size(2) == input2.size(1)  # M dimension
    
    B, N, M = input1.shape
    _, _, P = input2.shape
    
    # Create output tensor
    if out is None:
        out = torch.empty(B, N, P, dtype=input1.dtype, device=input1.device)
    else:
        assert out.shape == (B, N, P)
        assert out.dtype == input1.dtype
        assert out.device == input1.device
    
    # Launch kernel
    grid = (triton.cdiv(N, 16), triton.cdiv(P, 16), B)
    _fused_bmm_dropout_gelu_kernel[grid](
        input1, input2, out,
        None,  # dropout mask
        B, N, M, P,
        p,
        training,
        approximate,
        BLOCK_SIZE_M=16,
        BLOCK_SIZE_N=16,
        BLOCK_SIZE_K=16,
        BLOCK_SIZE_O=16
    )
    
    return out
