import torch
import triton
import triton.language as tl

@triton.jit
def adam_kernel(
    param_ptr, grad_ptr, exp_avg_ptr, exp_avg_sq_ptr, max_exp_avg_sq_ptr,
    lr, beta1, beta2, eps, weight_decay, step_size, amsgrad, maximize,
    num_elements, BLOCK_SIZE: tl.constexpr
):
    # Compute the global thread index
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < num_elements
    
    # Load data
    param = tl.load(param_ptr + offsets, mask=mask)
    grad = tl.load(grad_ptr + offsets, mask=mask)
    
    # Load momentum states
    exp_avg = tl.load(exp_avg_ptr + offsets, mask=mask)
    exp_avg_sq = tl.load(exp_avg_sq_ptr + offsets, mask=mask)
    
    # Update momentum states
    exp_avg = beta1 * exp_avg + (1 - beta1) * grad
    exp_avg_sq = beta2 * exp_avg_sq + (1 - beta2) * grad * grad
    
    # AMSGrad variant
    if amsgrad:
        max_exp_avg_sq = tl.load(max_exp_avg_sq_ptr + offsets, mask=mask)
        max_exp_avg_sq = tl.maximum(max_exp_avg_sq, exp_avg_sq)
        denom = tl.sqrt(max_exp_avg_sq) + eps
        tl.store(max_exp_avg_sq_ptr + offsets, max_exp_avg_sq, mask=mask)
    else:
        denom = tl.sqrt(exp_avg_sq) + eps
    
    # Compute step size
    step_size = step_size if not maximize else -step_size
    
    # Update parameters
    if weight_decay != 0:
        param = param - weight_decay * param
    
    param = param - step_size * exp_avg / denom
    
    # Store updated parameters and momentum states
    tl.store(param_ptr + offsets, param, mask=mask)
    tl.store(exp_avg_ptr + offsets, exp_avg, mask=mask)
    tl.store(exp_avg_sq_ptr + offsets, exp_avg_sq, mask=mask)

class Adam:
    def __init__(self, params, lr=1e-3, betas=(0.9, 0.999), eps=1e-8, weight_decay=0, amsgrad=False, foreach=None, maximize=False, capturable=False, differentiable=False, fused=None):
        self.params = list(params)
        self.lr = lr
        self.betas = betas
        self.eps = eps
        self.weight_decay = weight_decay
        self.amsgrad = amsgrad
        self.maximize = maximize
        self.capturable = capturable
        self.differentiable = differentiable
        self.fused = fused
        self.foreach = foreach
        self.state = {}
        self.step = 0
        
        # Initialize momentum states
        for i, param in enumerate(self.params):
            self.state[i] = {
                'exp_avg': torch.zeros_like(param),
                'exp_avg_sq': torch.zeros_like(param),
                'max_exp_avg_sq': torch.zeros_like(param) if amsgrad else None
            }
    
    def step(self):
        if self.fused:
            self._fused_step()
        else:
            self._basic_step()
    
    def _basic_step(self):
        self.step += 1
        beta1, beta2 = self.betas
        bias_correction1 = 1 - beta1 ** self.step
        bias_correction2 = 1 - beta2 ** self.step
        step_size = self.lr / bias_correction1 if not self.maximize else -self.lr / bias_correction1
        
        for i, param in enumerate(self.params):
            if param.grad is None:
                continue
                
            grad = param.grad
            exp_avg = self.state[i]['exp_avg']
            exp_avg_sq = self.state[i]['exp_avg_sq']
            max_exp_avg_sq = self.state[i]['max_exp_avg_sq']
            
            # Update momentum states
            exp_avg = beta1 * exp_avg + (1 - beta1) * grad
            exp_avg_sq = beta2 * exp_avg_sq + (1 - beta2) * grad * grad
            
            if self.amsgrad:
                max_exp_avg_sq = torch.maximum(max_exp_avg_sq, exp_avg_sq)
                denom = torch.sqrt(max_exp_avg_sq) + self.eps
            else:
                denom = torch.sqrt(exp_avg_sq) + self.eps
            
            # Update parameters
            if self.weight_decay != 0:
                param = param - self.weight_decay * param
            
            param = param - step_size * exp_avg / denom
            
            # Store updated states
            self.state[i]['exp_avg'] = exp_avg
            self.state[i]['exp_avg_sq'] = exp_avg_sq
            if self.amsgrad:
                self.state[i]['max_exp_avg_sq'] = max_exp_avg_sq
    
    def _fused_step(self):
        # Fused implementation using Triton
        self.step += 1
        beta1, beta2 = self.betas
        bias_correction1 = 1 - beta1 ** self.step
        bias_correction2 = 1 - beta2 ** self.step
        step_size = self.lr / bias_correction1 if not self.maximize else -self.lr / bias_correction1
        
        BLOCK_SIZE = 256
        num_elements = self.params[0].numel()
        
        for i, param in enumerate(self.params):
            if param.grad is None:
                continue
                
            grad = param.grad
            exp_avg = self.state[i]['exp_avg']
            exp_avg_sq = self.state[i]['exp_avg_sq']
            max_exp_avg_sq = self.state[i]['max_exp_avg_sq']
            
            # Launch Triton kernel
            grid = (triton.cdiv(num_elements, BLOCK_SIZE),)
            adam_kernel[grid](
                param.data_ptr(),
                grad.data_ptr(),
                exp_avg.data_ptr(),
                exp_avg_sq.data_ptr(),
                max_exp_avg_sq.data_ptr() if self.amsgrad else 0,
                self.lr,
                beta1,
                beta2,
                self.eps,
                self.weight_decay,
                step_size,
                self.amsgrad,
                self.maximize,
                num_elements,
                BLOCK_SIZE
            )
            
            # Update states
            self.state[i]['exp_avg'] = exp_avg
            self.state[i]['exp_avg_sq'] = exp_avg_sq
            if self.amsgrad:
                self.state[i]['max_exp_avg_sq'] = max_exp_avg_sq
