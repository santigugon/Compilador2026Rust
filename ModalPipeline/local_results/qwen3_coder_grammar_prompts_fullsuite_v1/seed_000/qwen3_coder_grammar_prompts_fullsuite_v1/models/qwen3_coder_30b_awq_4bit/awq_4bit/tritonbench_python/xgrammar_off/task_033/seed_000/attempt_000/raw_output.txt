```python
import torch
import triton
import triton.language as tl

@triton.jit
def logsumexp_kernel(
    input_ptr, 
    output_ptr, 
    num_elements, 
    dim_size, 
    stride_input, 
    stride_output,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < num_elements
    
    input_ptrs = input_ptr + offsets * stride_input
    input_vals = tl.load(input_ptrs, mask=mask)
    
    # Find max value for numerical stability
    max_val = tl.max(input_vals)
    
    # Compute sum of exponentials
    exp_vals = tl.exp(input_vals - max_val)
    sum_exp = tl.sum(exp_vals)
    
    # Compute log of sum of exponentials
    log_sum_exp = tl.log(sum_exp) + max_val
    
    # Store result
    output_ptrs = output_ptr + pid * stride_output
    tl.store(output_ptrs, log_sum_exp)

def logsumexp(input, dim, keepdim=False, *, out=None) -> torch.Tensor:
    if out is not None:
        raise NotImplementedError("out parameter is not supported")
    
    input = input.contiguous()
    input_shape = input.shape
    input_rank = len(input_shape)
    
    # Normalize dim to be in [0, input_rank)
    if dim < 0:
        dim = dim + input_rank
    
    # Create output tensor shape
    if keepdim:
        output_shape = list(input_shape)
        output_shape[dim] = 1
    else:
        output_shape = [input_shape[i] for i in range(input_rank) if i != dim]
    
    # Calculate output tensor
    output = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    
    # Get the size of the dimension we're reducing
    dim_size = input_shape[dim]
    
    # Calculate total elements
    num_elements = input.numel()
    
    # Set up kernel launch parameters
    BLOCK_SIZE = 1024
    num_blocks = (num_elements + BLOCK_SIZE - 1) // BLOCK_SIZE
    
    # Create a view of the input tensor for the kernel
    # We need to handle the reduction dimension specially
    if dim == input_rank - 1:
        stride_input = 1
        stride_output = 1
    else:
        stride_input = input.stride()[dim]
        stride_output = 1
    
    # Launch kernel
    logsumexp_kernel[(num_blocks,)](
        input.data_ptr(),
        output.data_ptr(),
        num_elements,
        dim_size,
        stride_input,
        stride_output,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    return output
```
