import torch
import triton
import triton.language as tl

@triton.jit
def logspace_kernel(
    start_ptr, end_ptr, out_ptr, steps, base, 
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < steps
    
    start_val = tl.load(start_ptr)
    end_val = tl.load(end_ptr)
    
    # Calculate log base^start to log base^end
    log_start = tl.log(start_val) / tl.log(base)
    log_end = tl.log(end_val) / tl.log(base)
    
    # Generate linearly spaced values in log space
    log_vals = tl.libdevice.log10(1.0 + (log_end - log_start) * offsets / (steps - 1))
    
    # Convert back to base^value
    result = tl.pow(base, log_vals)
    
    tl.store(out_ptr + offsets, result, mask=mask)

def logspace(start, end, steps, base=10.0, *, out=None, dtype=None, layout=torch.strided, device=None, requires_grad=False):
    if device is None:
        device = torch.get_default_device()
    
    if dtype is None:
        dtype = torch.get_default_dtype()
    
    if out is None:
        out = torch.empty(steps, dtype=dtype, device=device)
    else:
        if out.shape != (steps,):
            raise ValueError("out tensor must have shape (steps,)")
        if out.dtype != dtype:
            raise ValueError("out tensor must have the same dtype as requested")
        if out.device != device:
            raise ValueError("out tensor must be on the same device as requested")
    
    # Handle scalar inputs
    if not isinstance(start, torch.Tensor):
        start = torch.tensor(start, device=device, dtype=dtype)
    if not isinstance(end, torch.Tensor):
        end = torch.tensor(end, device=device, dtype=dtype)
    
    # Ensure tensors are 0-dimensional
    if start.dim() != 0:
        raise ValueError("start must be a 0-dimensional tensor")
    if end.dim() != 0:
        raise ValueError("end must be a 0-dimensional tensor")
    
    # Launch kernel
    BLOCK_SIZE = 256
    grid_size = (steps + BLOCK_SIZE - 1) // BLOCK_SIZE
    
    logspace_kernel[grid_size,](start, end, out, steps, base, BLOCK_SIZE)
    
    if requires_grad:
        out.requires_grad_(True)
    
    return out
