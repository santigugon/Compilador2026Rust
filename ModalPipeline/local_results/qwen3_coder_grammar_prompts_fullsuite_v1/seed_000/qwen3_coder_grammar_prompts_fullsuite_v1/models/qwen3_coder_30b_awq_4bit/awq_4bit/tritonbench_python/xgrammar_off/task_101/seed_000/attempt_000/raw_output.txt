import torch
import triton
import triton.language as tl

@triton.jit
def digamma_kernel(x_ptr, out_ptr, n_elements, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # Compute digamma using asymptotic expansion
    # For large x, digamma(x) ≈ ln(x) - 1/(2x) - 1/(12x^2) + 1/(120x^4) - 1/(252x^6)
    # For small x, use recurrence relation: digamma(x+1) = digamma(x) + 1/x
    
    # Handle special cases
    result = tl.where(x == 0.0, float('-inf'), tl.zeros_like(x))
    
    # For positive values, use asymptotic expansion
    x_positive = tl.where(x > 0.0, x, 1.0)
    x_reciprocal = 1.0 / x_positive
    
    # Asymptotic expansion coefficients
    x_squared = x_positive * x_positive
    x_fourth = x_squared * x_squared
    x_sixth = x_fourth * x_squared
    
    # Compute approximation: ln(x) - 1/(2x) - 1/(12x^2) + 1/(120x^4) - 1/(252x^6)
    term1 = tl.log(x_positive)
    term2 = -0.5 * x_reciprocal
    term3 = -1.0 / (12.0 * x_squared)
    term4 = 1.0 / (120.0 * x_fourth)
    term5 = -1.0 / (252.0 * x_sixth)
    
    asymptotic_result = term1 + term2 + term3 + term4 + term5
    
    # Handle small x values with recurrence relation
    # For x < 1, use digamma(x) = digamma(x+1) - 1/x
    x_small = tl.where(x < 1.0, x, 1.0)
    x_small_reciprocal = 1.0 / x_small
    x_small_plus_one = x_small + 1.0
    
    # Simple recurrence for small x
    recurrence_result = tl.where(x < 1.0, 
                                tl.load(digamma_kernel, x_small_plus_one) - x_small_reciprocal,
                                0.0)
    
    # Combine results
    final_result = tl.where(x > 0.0, 
                           tl.where(x < 1.0, recurrence_result, asymptotic_result),
                           result)
    
    tl.store(out_ptr + offsets, final_result, mask=mask)

def digamma(input, *, out=None):
    if out is None:
        out = torch.empty_like(input)
    else:
        if out.shape != input.shape:
            raise ValueError("Output tensor must have the same shape as input tensor")
