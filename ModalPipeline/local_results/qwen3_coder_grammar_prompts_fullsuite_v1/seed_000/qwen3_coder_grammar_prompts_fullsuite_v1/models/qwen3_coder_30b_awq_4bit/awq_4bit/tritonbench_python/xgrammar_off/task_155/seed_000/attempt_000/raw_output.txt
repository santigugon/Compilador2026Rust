import torch
import triton
import triton.language as tl
from typing import Union, Tuple, Optional

@triton.jit
def _conv2d_add_kernel(
    input_ptr, weight_ptr, bias_ptr, other_ptr, output_ptr,
    batch_size, in_channels, out_channels, iH, iW, oH, oW,
    kH, kW, stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w,
    groups, alpha,
    input_batch_stride, input_channel_stride, input_h_stride, input_w_stride,
    weight_batch_stride, weight_channel_stride, weight_h_stride, weight_w_stride,
    bias_batch_stride, other_batch_stride, other_channel_stride, other_h_stride, other_w_stride,
    output_batch_stride, output_channel_stride, output_h_stride, output_w_stride,
    BLOCK_SIZE: tl.constexpr
):
    batch_idx = tl.program_id(0)
    out_ch_idx = tl.program_id(1)
    
    # Each block handles one output channel
    if out_ch_idx >= out_channels:
        return
    
    # Initialize output
    output = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    
    # Loop over groups
    for g in range(groups):
        # Calculate group-specific indices
        group_in_ch = in_channels // groups
        group_out_ch = out_channels // groups
        
        # Calculate group-specific pointers
        input_base = input_ptr + batch_idx * input_batch_stride + g * group_in_ch * input_channel_stride
        weight_base = weight_ptr + out_ch_idx * weight_batch_stride + g * group_in_ch * weight_channel_stride
        
        # Convolution computation
        for kh in range(kH):
            for kw in range(kW):
                # Calculate input indices with padding and dilation
                ih_start = kh * dilation_h - pad_h
                iw_start = kw * dilation_w - pad_w
                
                # Loop over output spatial dimensions
                for oh in range(oH):
                    for ow in range(oW):
                        ih = ih_start + oh * stride_h
                        iw = iw_start + ow * stride_w
                        
                        # Check bounds
                        if ih >= 0 and ih < iH and iw >= 0 and iw < iW:
                            # Load input and weight
                            input_val = tl.load(input_base + ih * input_h_stride + iw * input_w_stride)
                            weight_val = tl.load(weight_base + kh * weight_h_stride + kw * weight_w_stride)
                            output += input_val * weight_val
    
    # Add bias if present
    if bias_ptr is not None:
        bias_val = tl.load(bias_ptr + out_ch_idx * bias_batch_stride)
        output += bias_val
    
    # Add other tensor or scalar
    if other_ptr is not None:
        other_val = tl.load(other_ptr + batch_idx * other_batch_stride + out_ch_idx * other_channel_stride)
        output += alpha * other_val
    
    # Store output
    output_base = output_ptr + batch_idx * output_batch_stride + out_ch_idx * output_channel_stride
    for oh in range(oH):
        for ow in range(oW):
            output_idx = oh * output_h_stride + ow * output_w_stride
            tl.store(output_base + output_idx, output[0])
