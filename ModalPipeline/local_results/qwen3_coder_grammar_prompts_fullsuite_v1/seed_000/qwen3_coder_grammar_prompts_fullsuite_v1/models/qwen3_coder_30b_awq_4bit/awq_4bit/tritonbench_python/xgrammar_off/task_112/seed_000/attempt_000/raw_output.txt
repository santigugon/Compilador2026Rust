import torch
import triton
import triton.language as tl

@triton.jit
def min_kernel(
    input_ptr, 
    output_ptr, 
    indices_ptr,
    n_rows, 
    n_cols,
    BLOCK_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    # Initialize min and index
    min_val = tl.full([1], float('inf'), dtype=tl.float32)
    min_idx = tl.full([1], 0, dtype=tl.int64)
    
    # Load data in chunks
    for col_start in range(0, n_cols, BLOCK_SIZE):
        col_offset = col_start + tl.arange(0, BLOCK_SIZE)
        mask = col_offset < n_cols
        
        # Load input values
        input_vals = tl.load(input_ptr + row_idx * n_cols + col_offset, mask=mask, other=float('inf'))
        
        # Find min and update indices
        current_min = tl.min(input_vals)
        current_min_idx = tl.argmin(input_vals)
        
        # Update global min and index
        if current_min < min_val:
            min_val = current_min
            min_idx = current_min_idx + col_start
    
    # Store results
    tl.store(output_ptr + row_idx, min_val)
    tl.store(indices_ptr + row_idx, min_idx)

def min(input, dim, keepdim=False, *, out=None):
    if dim < 0:
        dim = input.dim() + dim
    
    if dim >= input.dim():
        raise ValueError("Dimension out of range")
    
    # Get input shape
    shape = input.shape
    n_rows = 1
    n_cols = 1
    
    # Calculate dimensions
    for i in range(len(shape)):
        if i == dim:
            n_cols = shape[i]
        else:
            n_rows *= shape[i]
    
    # Create output tensors
    if out is not None:
        output = out[0]
        indices = out[1]
    else:
        output = torch.empty(n_rows, dtype=torch.float32, device=input.device)
        indices = torch.empty(n_rows, dtype=torch.int64, device=input.device)
    
    # Launch kernel
    if n_rows > 0:
        BLOCK_SIZE = 256
        grid = (n_rows, 1, 1)
        min_kernel[grid](
            input_ptr=input.data_ptr(),
            output_ptr=output.data_ptr(),
            indices_ptr=indices.data_ptr(),
            n_rows=n_rows,
            n_cols=n_cols,
            BLOCK_SIZE=BLOCK_SIZE
        )
    
    # Reshape outputs if needed
    if keepdim:
        new_shape = list(shape)
        new_shape[dim] = 1
        output = output.view(new_shape)
        indices = indices.view(new_shape)
    else:
        new_shape = list(shape)
        new_shape.pop(dim)
        output = output.view(new_shape)
        indices = indices.view(new_shape)
    
    return (output, indices)
