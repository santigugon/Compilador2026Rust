import torch
import triton
import triton.language as tl

@triton.jit
def elu_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_size, weight_size, bias_size,
    alpha, inplace,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < input_size
    
    # Load input
    input_vals = tl.load(input_ptr + offsets, mask=mask)
    
    # Linear transformation: input @ weight.T + bias
    # For simplicity, assuming input is 1D and weight is 1D (single output)
    # In practice, this would need to handle proper matrix multiplication
    weight_vals = tl.load(weight_ptr + offsets, mask=mask)
    linear_result = input_vals * weight_vals
    
    if bias_ptr is not None:
        bias_vals = tl.load(bias_ptr + offsets, mask=mask)
        linear_result = linear_result + bias_vals
    
    # Apply ELU activation
    elu_result = tl.where(linear_result > 0, linear_result, alpha * (tl.exp(linear_result) - 1))
    
    # Store result
    if inplace:
        tl.store(input_ptr + offsets, elu_result, mask=mask)
    else:
        tl.store(output_ptr + offsets, elu_result, mask=mask)

def elu_linear(input, weight, bias=None, alpha=1.0, inplace=False):
    # Ensure inputs are contiguous
    input = input.contiguous()
    weight = weight.contiguous()
    if bias is not None:
        bias = bias.contiguous()
    
    # Determine output size
    output_size = input.shape[0]  # Assuming single output for simplicity
    
    # Allocate output tensor if not inplace
    if not inplace:
        output = torch.empty_like(input)
    else:
        output = input
    
    # Launch kernel
    grid = (triton.cdiv(output_size, 1024),)
    elu_linear_kernel[grid](
        input_ptr=input.data_ptr(),
        weight_ptr=weight.data_ptr(),
        bias_ptr=bias.data_ptr() if bias is not None else None,
        output_ptr=output.data_ptr(),
        input_size=output_size,
        weight_size=weight.numel(),
        bias_size=bias.numel() if bias is not None else 0,
        alpha=alpha,
        inplace=inplace,
        BLOCK_SIZE=1024
    )
    
    return output
