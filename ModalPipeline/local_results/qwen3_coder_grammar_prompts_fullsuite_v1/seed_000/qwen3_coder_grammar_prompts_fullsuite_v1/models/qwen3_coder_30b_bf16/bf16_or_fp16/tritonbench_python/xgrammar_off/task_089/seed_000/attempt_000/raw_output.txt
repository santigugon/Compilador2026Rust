import torch
import triton
import triton.language as tl

@triton.jit
def cosine_similarity_kernel(
    x1_ptr, x2_ptr, out_ptr,
    x1_row_stride, x2_row_stride, out_row_stride,
    n_features, n_rows,
    eps,
    BLOCK_SIZE: tl.constexpr
):
    row = tl.program_id(0)
    if row >= n_rows:
        return
    
    x1_row = x1_ptr + row * x1_row_stride
    x2_row = x2_ptr + row * x2_row_stride
    
    # Compute dot product
    dot = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    x1_norm = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    x2_norm = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for i in range(0, n_features, BLOCK_SIZE):
        mask = tl.arange(0, BLOCK_SIZE) + i < n_features
        x1_vals = tl.load(x1_row + i, mask=mask, other=0.0)
        x2_vals = tl.load(x2_row + i, mask=mask, other=0.0)
        
        dot += x1_vals * x2_vals
        x1_norm += x1_vals * x1_vals
        x2_norm += x2_vals * x2_vals
    
    dot = tl.sum(dot)
    x1_norm = tl.sum(x1_norm)
    x2_norm = tl.sum(x2_norm)
    
    # Add epsilon to avoid division by zero
    x1_norm = tl.maximum(x1_norm, eps)
    x2_norm = tl.maximum(x2_norm, eps)
    
    # Compute cosine similarity
    norm = tl.sqrt(x1_norm) * tl.sqrt(x2_norm)
    similarity = dot / norm
    
    # Write result
    out = out_ptr + row * out_row_stride
    tl.store(out, similarity)

@triton.jit
def avg_pool2d_kernel(
    input_ptr, output_ptr,
    input_row_stride, input_col_stride,
    output_row_stride, output_col_stride,
    h_in, w_in, h_out, w_out,
    kernel_size, stride, padding,
    BLOCK_SIZE: tl.constexpr
):
    row = tl.program_id(0)
    col = tl.program_id(1)
    
    if row >= h_out or col >= w_out:
        return
    
    h_start = row * stride - padding
    w_start = col * stride - padding
    
    sum_val = tl.zeros([1], dtype=tl.float32)
    count = 0
    
    for kh in range(kernel_size):
        for kw in range(kernel_size):
            h = h_start + kh
            w = w_start + kw
            
            if h >= 0 and h < h_in and w >= 0 and w < w_in:
                input_idx = h * input_row_stride + w * input_col_stride
                sum_val += tl.load(input_ptr + input_idx)
                count += 1
    
    if count > 0:
        avg = sum_val / count
    else:
        avg = 0.0
    
    output_idx = row * output_row_stride + col * output_col_stride
    tl.store(output_ptr + output_idx, avg)

def fused_avg_pool2d_cosine_similarity(x1, x2, kernel_size, stride=None, padding=0, eps=1e-8):
    if stride is None:
        stride = kernel_size
    
    # Compute cosine similarity
    n_rows, n_features = x1.shape
    assert x1.shape == x2.shape, "x1 and x2 must have the same shape"
    
    # Allocate output for cosine similarity
    cos_sim = torch.empty(n_rows, dtype=torch.float32, device=x1.device)
    
    # Launch cosine similarity kernel
    BLOCK_SIZE = 1024
    grid = (n_rows, 1)
    cosine_similarity_kernel[grid](
        x1, x2, cos_sim,
        x1.stride(0), x2.stride(0), cos_sim.stride(0),
        n_features, n_rows,
        eps,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    # Add singleton dimension (unsqueeze)
    cos_sim = cos_sim.unsqueeze(1)  # Shape: [n_rows, 1]
    
    # Apply 2D average pooling
    # Reshape to 2D: [n_rows, 1] -> [n_rows, 1, 1, 1] for pooling
    # But we need to reshape to [n_rows, 1] to [1, 1, n_rows, 1] for pooling
    # Actually, we need to reshape to [1, 1, n_rows, 1] for proper pooling
    # But since we're doing 2D pooling on a 1D tensor, we need to think differently
    
    # Reshape to [1, 1, n_rows, 1] for pooling
    cos_sim_reshaped = cos_sim.view(1, 1, n_rows, 1)
    
    # Compute output dimensions
    h_out = (n_rows + 2 * padding - kernel_size) // stride + 1
    w_out = (1 + 2 * padding - kernel_size) // stride + 1
    
    # Ensure output dimensions are valid
    if h_out <= 0 or w_out <= 0:
        # Return zeros tensor
        return torch.zeros(1, 1, 1, 1, dtype=torch.float32, device=x1.device)
    
    # Allocate output for pooling
    pooled = torch.empty(1, 1, h_out, w_out, dtype=torch.float32, device=x1.device)
    
    # Launch pooling kernel
    grid = (h_out, w_out)
    avg_pool2d_kernel[grid](
        cos_sim_reshaped, pooled,
        cos_sim_reshaped.stride(0), cos_sim_reshaped.stride(1),
        pooled.stride(0), pooled.stride(1),
        n_rows, 1, h_out, w_out,
        kernel_size, stride, padding,
        BLOCK_SIZE=1024
    )
    
    # Return the final result
    return pooled
