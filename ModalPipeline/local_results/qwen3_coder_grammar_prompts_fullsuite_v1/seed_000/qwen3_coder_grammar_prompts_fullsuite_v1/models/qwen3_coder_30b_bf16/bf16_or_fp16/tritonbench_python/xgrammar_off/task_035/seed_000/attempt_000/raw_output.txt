import torch
import triton
import triton.language as tl

@triton.jit
def _fused_mv_sigmoid_sub_kernel(
    input_ptr, vec_ptr, other_ptr, out_ptr,
    n: tl.constexpr, m: tl.constexpr,
    alpha: tl.constexpr,
    input_stride_0: tl.constexpr, input_stride_1: tl.constexpr,
    vec_stride: tl.constexpr,
    other_stride: tl.constexpr,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    row = pid
    
    if row >= n:
        return
    
    # Load vector v
    vec = tl.load(vec_ptr + tl.arange(0, BLOCK_SIZE) * vec_stride, mask=tl.arange(0, BLOCK_SIZE) < m, other=0.0)
    
    # Compute dot product for this row
    acc = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    for i in range(0, m, BLOCK_SIZE):
        vec_offsets = i + tl.arange(0, BLOCK_SIZE)
        vec_mask = vec_offsets < m
        v = tl.load(vec_ptr + vec_offsets * vec_stride, mask=vec_mask, other=0.0)
        
        input_offsets = row * input_stride_0 + vec_offsets * input_stride_1
        input_mask = vec_mask
        a = tl.load(input_ptr + input_offsets, mask=input_mask, other=0.0)
        
        acc += a * v
    
    # Reduce to scalar
    result = tl.sum(acc, axis=0)
    
    # Apply sigmoid
    sigmoid_result = 1.0 / (1.0 + tl.exp(-result))
    
    # Load other (scalar or tensor)
    if other_ptr is not None:
        other = tl.load(other_ptr + tl.arange(0, 1) * other_stride, mask=tl.arange(0, 1) < 1, other=0.0)
        final_result = sigmoid_result - alpha * other
    else:
        final_result = sigmoid_result
    
    # Store result
    tl.store(out_ptr + row, final_result)

def fused_mv_sigmoid_sub(input, vec, other, alpha=1, *, out=None):
    # Validate inputs
    assert input.dim() == 2, "input must be a 2D tensor"
    assert vec.dim() == 1, "vec must be a 1D tensor"
    assert input.size(1) == vec.size(0), "input and vec dimensions must match"
    
    n, m = input.shape
    out = torch.empty(n, dtype=input.dtype, device=input.device) if out is None else out
    
    # Handle scalar other
    if not torch.is_tensor(other):
        other = torch.tensor(other, dtype=input.dtype, device=input.device)
    
    # Determine block size
    BLOCK_SIZE = 256
    grid = (n,)
    
    # Prepare strides
    input_stride_0 = input.stride(0)
    input_stride_1 = input.stride(1)
    vec_stride = vec.stride(0)
    other_stride = other.stride(0) if other.dim() > 0 else 1
    
    # Launch kernel
    _fused_mv_sigmoid_sub_kernel[grid](
        input, vec, other, out,
        n, m,
        alpha,
        input_stride_0, input_stride_1,
        vec_stride,
        other_stride,
        BLOCK_SIZE
    )
    
    return out
