import torch
import triton
import triton.language as tl

@triton.jit
def add_kernel(
    input_ptr, other_ptr, output_ptr,
    input_size, other_size, output_size,
    alpha,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < output_size
    
    # Load input tensor
    input_vals = tl.load(input_ptr + offsets, mask=mask)
    
    # Load other tensor or scalar
    if other_size == 1:
        other_val = tl.load(other_ptr)
        other_vals = other_val
    else:
        other_vals = tl.load(other_ptr + offsets, mask=mask)
    
    # Compute result
    result = input_vals + alpha * other_vals
    
    # Store result
    tl.store(output_ptr + offsets, result, mask=mask)

def add(input, other, *, alpha=1, out=None):
    # Handle scalar other
    if not isinstance(other, torch.Tensor):
        other = torch.tensor(other, dtype=input.dtype, device=input.device)
    
    # Ensure both tensors are on the same device
    if other.device != input.device:
        other = other.to(input.device)
    
    # Handle broadcasting
    input_shape = input.shape
    other_shape = other.shape
    
    # Compute output shape
    max_dims = max(len(input_shape), len(other_shape))
    input_shape = [1] * (max_dims - len(input_shape)) + list(input_shape)
    other_shape = [1] * (max_dims - len(other_shape)) + list(other_shape)
    
    output_shape = []
    for i in range(max_dims):
        if input_shape[i] == 1:
            output_shape.append(other_shape[i])
        elif other_shape[i] == 1:
            output_shape.append(input_shape[i])
        else:
            assert input_shape[i] == other_shape[i], "Shapes are not broadcastable"
            output_shape.append(input_shape[i])
    
    output_size = 1
    for dim in output_shape:
        output_size *= dim
    
    # Create output tensor
    if out is None:
        output = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    else:
        output = out
    
    # Launch kernel
    if output_size > 0:
        BLOCK_SIZE = 1024
        grid_size = (output_size + BLOCK_SIZE - 1) // BLOCK_SIZE
        
        # Flatten tensors for kernel
        input_flat = input.reshape(-1)
        other_flat = other.reshape(-1)
        output_flat = output.reshape(-1)
        
        add_kernel[grid_size](
            input_flat, other_flat, output_flat,
            input_flat.numel(), other_flat.numel(), output_flat.numel(),
            alpha,
            BLOCK_SIZE
        )
    
    return output
