import torch
import triton
import triton.language as tl

@triton.jit
def _fused_gather_masked_fill_kernel(
    input_ptr, index_ptr, mask_ptr, out_ptr,
    input_strides, index_strides, mask_strides, out_strides,
    input_shape, index_shape, mask_shape, out_shape,
    dim: tl.constexpr,
    value: tl.constexpr,
    num_elements: tl.constexpr,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < num_elements
    
    # Convert linear offset to multi-dimensional indices
    indices = []
    temp_offset = offsets
    for i in range(len(out_shape) - 1, -1, -1):
        if i == dim:
            # For the gather dimension, we need to use the index tensor
            idx_offset = temp_offset // out_strides[i] if i < len(out_strides) - 1 else temp_offset
            idx_offset = idx_offset % index_shape[i]
            indices.insert(0, idx_offset)
        else:
            # For other dimensions, we can directly compute the index
            idx_offset = temp_offset // out_strides[i] if i < len(out_strides) - 1 else temp_offset
            idx_offset = idx_offset % out_shape[i]
            indices.insert(0, idx_offset)
        temp_offset = temp_offset % out_strides[i] if i > 0 else temp_offset
    
    # Compute linear indices for input tensor
    input_idx = 0
    for i in range(len(input_shape)):
        if i == dim:
            # Use index tensor to get the actual index
            idx_val = tl.load(index_ptr + indices[i], mask=mask, other=0)
            input_idx += idx_val * input_strides[i]
        else:
            input_idx += indices[i] * input_strides[i]
    
    # Load input value
    input_val = tl.load(input_ptr + input_idx, mask=mask, other=0.0)
    
    # Check mask
    mask_val = tl.load(mask_ptr + offsets, mask=mask, other=False)
    
    # Apply masked fill
    result = tl.where(mask_val, value, input_val)
    
    # Store result
    tl.store(out_ptr + offsets, result, mask=mask)

def fused_gather_masked_fill(input, dim, index, mask, value, *, sparse_grad=False, out=None):
    # Validate inputs
    if not torch.is_tensor(input):
        raise TypeError("input must be a tensor")
    if not torch.is_tensor(index):
        raise TypeError("index must be a tensor")
    if not torch.is_tensor(mask):
        raise TypeError("mask must be a tensor")
    
    # Handle negative dimension
    if dim < 0:
        dim = input.dim() + dim
    
    # Validate dimension
    if dim < 0 or dim >= input.dim():
        raise ValueError(f"dim {dim} out of range for input tensor with {input.dim()} dimensions")
    
    # Check if index has the same shape as output
    out_shape = list(input.shape)
    out_shape[dim] = index.shape[dim]
    out_shape = tuple(out_shape)
    
    # Check if mask is broadcastable to output shape
    if not mask.shape == out_shape:
        # Try to broadcast mask to output shape
        try:
            torch.broadcast_tensors(mask, torch.empty(out_shape))
        except RuntimeError:
            raise ValueError("mask is not broadcastable to the output shape")
    
    # Create output tensor
    if out is None:
        out = torch.empty(out_shape, dtype=input.dtype, device=input.device)
    else:
        if out.shape != out_shape:
            raise ValueError(f"out tensor must have shape {out_shape}")
        if out.dtype != input.dtype:
            raise ValueError(f"out tensor must have dtype {input.dtype}")
        if out.device != input.device:
            raise ValueError("out tensor must be on the same device as input tensor")
    
    # Handle scalar value
    if not isinstance(value, (int, float)):
        raise TypeError("value must be a scalar")
    
    # Handle sparse_grad - this is just a parameter for compatibility, not implemented in kernel
    if sparse_grad:
        # In practice, sparse_grad would affect gradient computation, but not the forward pass
        pass
    
    # Flatten tensors for kernel processing
    input_flat = input.flatten()
    index_flat = index.flatten()
    mask_flat = mask.flatten()
    out_flat = out.flatten()
    
    # Compute strides
    input_strides = [input.stride(i) for i in range(input.dim())]
    index_strides = [index.stride(i) for i in range(index.dim())]
    mask_strides = [mask.stride(i) for i in range(mask.dim())]
    out_strides = [out.stride(i) for i in range(out.dim())]
    
    # Compute shapes
    input_shape = input.shape
    index_shape = index.shape
    mask_shape = mask.shape
    out_shape = out.shape
    
    # Compute total number of elements
    num_elements = out.numel()
    
    # Launch kernel
    if num_elements == 0:
        return out
    
    block = 256
    grid = (triton.cdiv(num_elements, block),)
    
    # Create a temporary tensor for the kernel to work with
    _fused_gather_masked_fill_kernel[grid](
        input_flat, index_flat, mask_flat, out_flat,
        input_strides, index_strides, mask_strides, out_strides,
        input_shape, index_shape, mask_shape, out_shape,
        dim,
        value,
        num_elements,
        BLOCK=block
    )
    
    return out
