import torch
import triton
import triton.language as tl

@triton.jit
def fused_layer_norm_relu_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_row_stride, weight_row_stride, weight_col_stride,
    output_row_stride, output_col_stride,
    n_rows, n_cols, out_features,
    normalized_shape, eps,
    elementwise_affine: tl.constexpr,
    BLOCK_SIZE: tl.constexpr
):
    row = tl.program_id(0)
    if row >= n_rows:
        return
    
    # Load input row
    input_row = tl.load(input_ptr + row * input_row_stride + tl.arange(0, BLOCK_SIZE), mask=tl.arange(0, BLOCK_SIZE) < n_cols)
    
    # Linear transformation
    linear_output = tl.zeros((out_features,), dtype=tl.float32)
    for i in range(0, n_cols, BLOCK_SIZE):
        mask = (tl.arange(0, BLOCK_SIZE) + i) < n_cols
        input_vals = tl.load(input_row + i, mask=mask)
        for j in range(out_features):
            weight_vals = tl.load(weight_ptr + j * weight_row_stride + i, mask=mask)
            linear_output[j] += tl.sum(input_vals * weight_vals)
    
    # Add bias
    if bias_ptr is not None:
        bias_vals = tl.load(bias_ptr + tl.arange(0, out_features))
        linear_output += bias_vals
    
    # Apply ReLU
    linear_output = tl.where(linear_output > 0, linear_output, 0.0)
    
    # Layer normalization
    if elementwise_affine:
        # Compute mean and variance
        mean = tl.sum(linear_output) / normalized_shape
        var = tl.sum((linear_output - mean) * (linear_output - mean)) / normalized_shape
        inv_std = 1.0 / tl.sqrt(var + eps)
        
        # Normalize
        normalized = (linear_output - mean) * inv_std
        
        # Apply affine transformation (scale and shift)
        weight_norm = tl.load(weight_ptr + tl.arange(0, out_features))  # This is a simplification
        bias_norm = tl.load(bias_ptr + tl.arange(0, out_features))      # This is a simplification
        output = normalized * weight_norm + bias_norm
    else:
        # Compute mean and variance
        mean = tl.sum(linear_output) / normalized_shape
        var = tl.sum((linear_output - mean) * (linear_output - mean)) / normalized_shape
        inv_std = 1.0 / tl.sqrt(var + eps)
        
        # Normalize
        output = (linear_output - mean) * inv_std
    
    # Store output
    tl.store(output_ptr + row * output_row_stride + tl.arange(0, out_features), output)

def fused_layer_norm_relu_linear(input, weight, bias=None, normalized_shape=None, eps=1e-5, elementwise_affine=True):
    if normalized_shape is None:
        normalized_shape = weight.shape[0]
    if not isinstance(normalized_shape, (int, list, tuple)):
        normalized_shape = normalized_shape[-1]
    if isinstance(normalized_shape, (list, tuple)):
        normalized_shape = normalized_shape[-1]
    
    # Ensure input is contiguous
    input = input.contiguous()
    weight = weight.contiguous()
    if bias is not None:
        bias = bias.contiguous()
    
    # Get dimensions
    n_rows, n_cols = input.shape
    out_features = weight.shape[0]
    
    # Create output tensor
    output = torch.empty(n_rows, out_features, dtype=input.dtype, device=input.device)
    
    # Define block size
    BLOCK_SIZE = 1024
    
    # Launch kernel
    grid = (n_rows, 1, 1)
    fused_layer_norm_relu_linear_kernel[grid](
        input_ptr=input.data_ptr(),
        weight_ptr=weight.data_ptr(),
        bias_ptr=bias.data_ptr() if bias is not None else 0,
        output_ptr=output.data_ptr(),
        input_row_stride=n_cols,
        weight_row_stride=weight.shape[1],
        weight_col_stride=1,
        output_row_stride=out_features,
        output_col_stride=1,
        n_rows=n_rows,
        n_cols=n_cols,
        out_features=out_features,
        normalized_shape=normalized_shape,
        eps=eps,
        elementwise_affine=elementwise_affine,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    return output
