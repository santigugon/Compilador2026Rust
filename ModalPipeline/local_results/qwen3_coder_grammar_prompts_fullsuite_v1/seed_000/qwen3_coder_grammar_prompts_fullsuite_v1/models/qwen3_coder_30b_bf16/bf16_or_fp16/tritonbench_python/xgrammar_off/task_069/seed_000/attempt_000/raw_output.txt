import torch
import triton
import triton.language as tl

@triton.jit
def _fused_layer_norm_relu_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_row_stride, weight_row_stride, weight_col_stride,
    output_row_stride, output_col_stride,
    n_features: tl.constexpr, n_out: tl.constexpr,
    eps: tl.constexpr, BLOCK_SIZE: tl.constexpr
):
    # Get the row index
    row_idx = tl.program_id(0)
    
    # Load input row
    input_offsets = row_idx * input_row_stride + tl.arange(0, BLOCK_SIZE)
    input_row = tl.load(input_ptr + input_offsets, mask=input_offsets < n_features, other=0.0)
    
    # Compute linear transformation: y = x @ weight.T + bias
    linear_result = tl.zeros((n_out,), dtype=tl.float32)
    for i in range(0, n_features, BLOCK_SIZE):
        # Load weight column slice
        weight_offsets = i + tl.arange(0, BLOCK_SIZE)
        weight_slice = tl.load(weight_ptr + weight_offsets * weight_col_stride, mask=weight_offsets < n_features, other=0.0)
        
        # Compute dot product with input row
        for j in range(n_out):
            linear_result[j] += tl.sum(input_row * weight_slice * tl.load(weight_ptr + j * weight_row_stride + weight_offsets, mask=weight_offsets < n_features, other=0.0))
    
    # Add bias
    if bias_ptr is not None:
        bias_offsets = tl.arange(0, n_out)
        bias = tl.load(bias_ptr + bias_offsets, mask=bias_offsets < n_out, other=0.0)
        linear_result += bias
    
    # Apply ReLU
    linear_result = tl.where(linear_result > 0, linear_result, 0.0)
    
    # Compute layer normalization
    # Compute mean
    mean = tl.sum(linear_result) / n_out
    
    # Compute variance
    diff = linear_result - mean
    variance = tl.sum(diff * diff) / n_out
    
    # Normalize
    normalized = (linear_result - mean) / tl.sqrt(variance + eps)
    
    # Store result
    output_offsets = row_idx * output_row_stride + tl.arange(0, n_out)
    tl.store(output_ptr + output_offsets, normalized, mask=output_offsets < n_out)

def fused_layer_norm_relu_linear(input, weight, bias=None, normalized_shape=None, eps=1e-5, elementwise_affine=True):
    # Handle normalized_shape
    if normalized_shape is None:
        normalized_shape = weight.shape[0]
    if isinstance(normalized_shape, int):
        normalized_shape = [normalized_shape]
    
    # Validate shapes
    assert input.dim() >= 2, "Input must have at least 2 dimensions"
    assert weight.dim() == 2, "Weight must be 2D"
    assert weight.shape[1] == input.shape[-1], "Weight input dimension must match input last dimension"
    assert weight.shape[0] == normalized_shape[0], "Weight output dimension must match normalized_shape"
    
    if bias is not None:
        assert bias.shape[0] == weight.shape[0], "Bias must have same size as weight output dimension"
    
    # Prepare output tensor
    output_shape = list(input.shape[:-1]) + [weight.shape[0]]
    out = torch.empty(output_shape, dtype=input.dtype, device=input.device)
    
    # Get dimensions
    batch_size = input.numel() // input.shape[-1]
    n_features = input.shape[-1]
    n_out = weight.shape[0]
    
    # Launch kernel
    BLOCK_SIZE = 256
    grid = (batch_size,)
    
    # Prepare pointers
    input_ptr = input.data_ptr()
    weight_ptr = weight.data_ptr()
    bias_ptr = bias.data_ptr() if bias is not None else None
    output_ptr = out.data_ptr()
    
    # Compute strides
    input_row_stride = input.stride(-2) if input.dim() > 1 else 1
    weight_row_stride = weight.stride(0)
    weight_col_stride = weight.stride(1)
    output_row_stride = out.stride(-2) if out.dim() > 1 else 1
    output_col_stride = out.stride(-1)
    
    _fused_layer_norm_relu_linear_kernel[grid](
        input_ptr, weight_ptr, bias_ptr, output_ptr,
        input_row_stride, weight_row_stride, weight_col_stride,
        output_row_stride, output_col_stride,
        n_features, n_out, eps, BLOCK_SIZE
    )
    
    return out
