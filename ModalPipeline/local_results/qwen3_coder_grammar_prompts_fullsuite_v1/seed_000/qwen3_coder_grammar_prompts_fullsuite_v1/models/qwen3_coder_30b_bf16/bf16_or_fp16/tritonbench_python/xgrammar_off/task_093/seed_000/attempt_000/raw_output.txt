import triton
import triton.language as tl

@triton.jit
def _softmax_log_kernel(x_ptr, out_ptr, dim_size: tl.constexpr, n_elements: tl.constexpr, 
                       stride_x_row, stride_x_col, stride_out_row, stride_out_col, 
                       BLOCK: tl.constexpr):
    # Get the row index
    row_idx = tl.program_id(0)
    
    # Calculate the starting offset for this row
    x_row_start = row_idx * stride_x_row
    out_row_start = row_idx * stride_out_row
    
    # Load the row data
    offsets = tl.arange(0, BLOCK) * stride_x_col
    mask = offsets < n_elements
    
    # Load input values
    x = tl.load(x_ptr + x_row_start + offsets, mask=mask, other=0.0)
    
    # Apply log
    x = tl.log(x)
    
    # Apply softmax
    # Find max value for numerical stability
    max_val = tl.max(x, axis=0)
    x = x - max_val
    
    # Compute exp and sum
    x_exp = tl.exp(x)
    sum_exp = tl.sum(x_exp, axis=0)
    
    # Normalize
    x_softmax = x_exp / sum_exp
    
    # Store result
    tl.store(out_ptr + out_row_start + offsets, x_softmax, mask=mask)
