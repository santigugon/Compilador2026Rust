import torch
import triton
import triton.language as tl

@triton.jit
def _tanh_linear_kernel(input_ptr, weight_ptr, bias_ptr, output_ptr, 
                        in_features: tl.constexpr, out_features: tl.constexpr, 
                        batch_size: tl.constexpr, BLOCK_SIZE: tl.constexpr):
    # Get the batch dimension
    batch_idx = tl.program_id(0)
    
    # Load weight matrix (out_features, in_features)
    weight_row = tl.load(weight_ptr + tl.arange(0, in_features) + 
                        batch_idx * in_features * out_features, 
                        mask=tl.arange(0, in_features) < in_features, other=0.0)
    
    # Load input (in_features)
    input_row = tl.load(input_ptr + tl.arange(0, in_features) + 
                       batch_idx * in_features, 
                       mask=tl.arange(0, in_features) < in_features, other=0.0)
    
    # Compute linear transformation: input @ weight.T + bias
    linear_result = tl.sum(input_row * weight_row, axis=0)
    
    # Add bias if provided
    if bias_ptr is not None:
        bias_val = tl.load(bias_ptr + tl.arange(0, out_features), 
                          mask=tl.arange(0, out_features) < out_features, other=0.0)
        linear_result = linear_result + bias_val
    
    # Apply tanh activation
    output_val = 2.0 / (1.0 + tl.exp(-2.0 * linear_result)) - 1.0
    
    # Store result
    tl.store(output_ptr + tl.arange(0, out_features) + 
             batch_idx * out_features, 
             output_val, 
             mask=tl.arange(0, out_features) < out_features)

def tanh_linear(input, weight, bias=None):
    # Handle the case where input is 1D (batch_size=1)
    if input.dim() == 1:
        input = input.unsqueeze(0)
    
    batch_size = input.shape[0]
    in_features = input.shape[1]
    out_features = weight.shape[0]
    
    # Create output tensor
    output = torch.empty(batch_size, out_features, device=input.device, dtype=input.dtype)
    
    # Launch kernel
    block_size = 256
    grid_size = batch_size
    
    # Determine if bias is provided
    bias_ptr = bias if bias is not None else None
    
    _tanh_linear_kernel[grid_size](
        input, weight, bias_ptr, output,
        in_features, out_features, batch_size, 
        BLOCK_SIZE=block_size
    )
    
    # Remove batch dimension if input was 1D
    if input.shape[0] == 1:
        output = output.squeeze(0)
    
    return output
