import torch
import triton
import triton.language as tl

@triton.jit
def _pixel_shuffle_conv2d_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_height, input_width, output_height, output_width,
    in_channels, out_channels, kernel_h, kernel_w,
    stride_h, stride_w, padding_h, padding_w, dilation_h, dilation_w,
    groups, upscale_factor,
    input_batch, BLOCK_SIZE: tl.constexpr
):
    # Get program ID
    batch_id = tl.program_id(0)
    out_ch_id = tl.program_id(1)
    
    # Calculate output dimensions
    out_h = output_height
    out_w = output_width
    
    # Calculate group size
    group_size = in_channels // groups
    
    # Calculate output spatial indices
    out_y = tl.program_id(2)
    out_x = tl.program_id(3)
    
    # Calculate input spatial indices
    in_y = out_y * stride_h - padding_h
    in_x = out_x * stride_w - padding_w
    
    # Calculate output channel indices
    out_ch = out_ch_id * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask_ch = out_ch < out_channels
    
    # Calculate input channel indices
    in_ch = tl.arange(0, group_size)
    
    # Initialize accumulator
    acc = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    
    # Perform convolution
    for kh in range(kernel_h):
        for kw in range(kernel_w):
            # Calculate input coordinates
            in_y_k = in_y + kh * dilation_h
            in_x_k = in_x + kw * dilation_w
            
            # Check bounds
            in_y_valid = (in_y_k >= 0) & (in_y_k < input_height)
            in_x_valid = (in_x_k >= 0) & (in_x_k < input_width)
            valid = in_y_valid & in_x_valid
            
            # Load input data
            input_idx = batch_id * (in_channels * input_height * input_width) + \
                       (out_ch_id // (out_channels // groups)) * (group_size * input_height * input_width) + \
                       in_ch * (input_height * input_width) + \
                       in_y_k * input_width + in_x_k
            input_data = tl.load(input_ptr + input_idx, mask=valid & (in_ch < group_size), other=0.0)
            
            # Load weight data
            weight_idx = out_ch * (group_size * kernel_h * kernel_w) + \
                        in_ch * (kernel_h * kernel_w) + \
                        kh * kernel_w + kw
            weight_data = tl.load(weight_ptr + weight_idx, mask=mask_ch & (in_ch < group_size), other=0.0)
            
            # Accumulate
            acc += tl.sum(input_data * weight_data, axis=0)
    
    # Add bias
    if bias_ptr is not None:
        bias_idx = out_ch
        bias_data = tl.load(bias_ptr + bias_idx, mask=mask_ch, other=0.0)
        acc += bias_data
    
    # Store result
    output_idx = batch_id * (out_channels * out_h * out_w) + \
                out_ch_id * (out_h * out_w) + \
                out_y * out_w + out_x
    tl.store(output_ptr + output_idx, acc, mask=mask_ch)

def pixel_shuffle_conv2d(input: torch.Tensor, weight: torch.Tensor, bias=None, stride=1, padding=0, dilation=1, groups=1, upscale_factor=2) -> torch.Tensor:
    # Get input dimensions
    batch_size, in_channels, input_height, input_width = input.shape
    out_channels, _, kernel_h, kernel_w = weight.shape
    
    # Calculate output dimensions
    output_height = (input_height + 2 * padding - (dilation * (kernel_h - 1) + 1)) // stride + 1
    output_width = (input_width + 2 * padding - (dilation * (kernel_w - 1) + 1)) // stride + 1
    
    # Create output tensor
    output = torch.empty(
        batch_size, out_channels, output_height, output_width,
        dtype=input.dtype, device=input.device
    )
    
    # Handle bias
    if bias is not None:
        bias = bias.to(input.dtype)
    
    # Set up grid
    grid = (
        batch_size,
        triton.cdiv(out_channels, 32),
        output_height,
        output_width
    )
    
    # Launch kernel
    _pixel_shuffle_conv2d_kernel[grid](
        input, weight, bias, output,
        input_height, input_width, output_height, output_width,
        in_channels, out_channels, kernel_h, kernel_w,
        stride, stride, padding, padding, dilation, dilation,
        groups, upscale_factor,
        batch_size,
        BLOCK_SIZE=32
    )
    
    # Apply pixel shuffle
    if upscale_factor > 1:
        # Reshape to separate spatial and channel dimensions
        output = output.view(batch_size, out_channels, output_height, output_width)
        
        # Apply pixel shuffle
        # Rearrange to (batch, out_channels * upscale_factor * upscale_factor, output_height // upscale_factor, output_width // upscale_factor)
        output = output.contiguous()
        
        # For simplicity, we'll use PyTorch's pixel_shuffle if available
        # But since we're implementing in Triton, we'll do it manually
        # This is a simplified version - in practice, you'd want to do this in a more optimized way
        if out_channels % (upscale_factor * upscale_factor) == 0:
            channels_per_group = out_channels // (upscale_factor * upscale_factor)
            output = output.view(batch_size, channels_per_group, upscale_factor, upscale_factor, output_height, output_width)
            output = output.permute(0, 1, 4, 2, 5, 3).contiguous()
            output = output.view(batch_size, out_channels, output_height, output_width)
    
    return output
