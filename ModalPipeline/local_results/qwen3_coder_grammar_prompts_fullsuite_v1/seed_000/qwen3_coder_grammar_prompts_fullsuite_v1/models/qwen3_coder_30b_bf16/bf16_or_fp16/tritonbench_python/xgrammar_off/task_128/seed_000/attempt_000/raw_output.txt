import torch
import triton
import triton.language as tl

@triton.jit
def fused_bmm_rmsnorm_gelu_dropout_kernel(
    input1_ptr, input2_ptr, output_ptr, 
    rms_norm_weight_ptr, dropout_mask_ptr,
    B, N, M, P, 
    normalized_shape,
    dropout_p,
    eps,
    training,
    approximate,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
    BLOCK_SIZE_NORM: tl.constexpr
):
    # Get block indices
    batch_idx = tl.program_id(0)
    m_idx = tl.program_id(1)
    n_idx = tl.program_id(2)
    
    # Load input1 and input2
    input1_block_ptr = tl.make_block_ptr(
        input1_ptr, 
        shape=(B, N, M), 
        strides=(N * M, M, 1),
        offsets=(batch_idx, m_idx * BLOCK_SIZE_M, 0),
        block_shape=(BLOCK_SIZE_M, BLOCK_SIZE_K),
        order=(1, 0)
    )
    
    input2_block_ptr = tl.make_block_ptr(
        input2_ptr, 
        shape=(B, M, P), 
        strides=(M * P, P, 1),
        offsets=(batch_idx, 0, n_idx * BLOCK_SIZE_N),
        block_shape=(BLOCK_SIZE_K, BLOCK_SIZE_N),
        order=(1, 0)
    )
    
    # Compute BMM
    acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, M, BLOCK_SIZE_K):
        input1 = tl.load(input1_block_ptr, boundary_check=(0, 1))
        input2 = tl.load(input2_block_ptr, boundary_check=(0, 1))
        acc += tl.dot(input1, input2)
        input1_block_ptr = tl.advance(input1_block_ptr, (0, BLOCK_SIZE_K))
        input2_block_ptr = tl.advance(input2_block_ptr, (BLOCK_SIZE_K, 0))
    
    # Apply RMS normalization
    # Compute mean of squares
    norm_block_ptr = tl.make_block_ptr(
        input1_ptr, 
        shape=(B, N, M), 
        strides=(N * M, M, 1),
        offsets=(batch_idx, m_idx * BLOCK_SIZE_M, 0),
        block_shape=(BLOCK_SIZE_M, BLOCK_SIZE_NORM),
        order=(1, 0)
    )
    
    # Compute sum of squares for normalization
    sum_squares = tl.zeros((BLOCK_SIZE_M,), dtype=tl.float32)
    for k in range(0, normalized_shape, BLOCK_SIZE_NORM):
        norm_data = tl.load(norm_block_ptr, boundary_check=(0, 1))
        sum_squares += tl.sum(norm_data * norm_data, axis=1)
        norm_block_ptr = tl.advance(norm_block_ptr, (0, BLOCK_SIZE_NORM))
    
    # Compute RMS
    rms = tl.sqrt(sum_squares / normalized_shape + eps)
    
    # Normalize
    norm_weight = tl.load(rms_norm_weight_ptr + m_idx * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M))
    acc = acc / rms[:, None] * norm_weight[:, None]
    
    # Apply GELU
    if approximate == "tanh":
        # Approximate GELU with tanh
        x = acc
        acc = 0.5 * x * (1 + tl.tanh(0.7978845608028654 * (x + 0.044715 * x * x * x)))
    else:
        # Exact GELU
        acc = 0.5 * acc * (1 + tl.erf(acc / tl.sqrt(2.0)))
    
    # Apply dropout
    if training:
        # Generate random mask
        mask = tl.rand(tl.program_id(0) * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M), 
                       tl.program_id(1) * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) > dropout_p
        acc = acc * mask / (1.0 - dropout_p)
    
    # Store output
    output_block_ptr = tl.make_block_ptr(
        output_ptr, 
        shape=(B, N, P), 
        strides=(N * P, P, 1),
        offsets=(batch_idx, m_idx * BLOCK_SIZE_M, n_idx * BLOCK_SIZE_N),
        block_shape=(BLOCK_SIZE_M, BLOCK_SIZE_N),
        order=(1, 0)
    )
    tl.store(output_block_ptr, acc, boundary_check=(0, 1))

def fused_bmm_rmsnorm_gelu_dropout(input1, input2, normalized_shape, dropout_p=0.1, eps=1e-5, training=True, approximate='none', *, out=None):
    # Validate inputs
    assert input1.dim() == 3 and input2.dim() == 3
    assert input1.shape[0] == input2.shape[0]  # batch size
    assert input1.shape[2] == input2.shape[1]  # inner dimension
    assert input1.shape[0] == input2.shape[0]  # batch size
    
    B, N, M = input1.shape
    _, _, P = input2.shape
    
    # Create output tensor if not provided
    if out is None:
        out = torch.empty_like(input1[:, :, :P])
    
    # Create RMS normalization weight
    if isinstance(normalized_shape, int):
        norm_shape = normalized_shape
    else:
        norm_shape = normalized_shape[-1] if len(normalized_shape) > 0 else 1
    
    rms_norm_weight = torch.ones(norm_shape, dtype=input1.dtype, device=input1.device)
    
    # Create dropout mask
    dropout_mask = torch.empty(B, N, P, dtype=torch.bool, device=input1.device)
    
    # Launch kernel
    grid = (B, triton.cdiv(N, 128), triton.cdiv(P, 128))
    fused_bmm_rmsnorm_gelu_dropout_kernel[grid](
        input1, input2, out,
        rms_norm_weight, dropout_mask,
        B, N, M, P,
        norm_shape,
        dropout_p,
        eps,
        training,
        approximate,
        BLOCK_SIZE_M=128,
        BLOCK_SIZE_N=128,
        BLOCK_SIZE_K=128,
        BLOCK_SIZE_NORM=128
    )
    
    return out
