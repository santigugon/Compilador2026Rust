import torch
import triton
import triton.language as tl

@triton.jit
def cholesky_kernel(A_ptr, out_ptr, n, batch_size, stride_row, stride_col, BLOCK_SIZE: tl.constexpr):
    batch_idx = tl.program_id(0)
    tid = tl.program_id(1)
    
    # Load matrix A for this batch
    A = tl.load(A_ptr + batch_idx * stride_row * n + tid * stride_row + tl.arange(0, BLOCK_SIZE) * stride_col)
    
    # Initialize output matrix
    out = tl.zeros((BLOCK_SIZE, BLOCK_SIZE), dtype=tl.float32)
    
    # Cholesky decomposition
    for k in range(n):
        if tid == k:
            # Compute diagonal element
            sum_val = tl.sum(out[:, k] * out[:, k])
            out[k, k] = tl.sqrt(A[k * stride_row + k] - sum_val)
        elif tid > k:
            # Compute off-diagonal elements
            sum_val = tl.sum(out[k, :] * out[tid, :])
            out[tid, k] = (A[tid * stride_row + k] - sum_val) / out[k, k]
        else:
            # Set upper triangle to zero
            out[tid, k] = 0.0
    
    # Store result
    tl.store(out_ptr + batch_idx * stride_row * n + tid * stride_row + tl.arange(0, BLOCK_SIZE) * stride_col, out)

def linalg.cholesky(A, *, upper=False, out=None) -> torch.Tensor:
    if A.dtype not in [torch.float32, torch.float64, torch.complex64, torch.complex128]:
        raise ValueError("Only float32, float64, complex64, and complex128 dtypes are supported")
    
    if A.dim() < 2:
        raise ValueError("Input tensor must have at least 2 dimensions")
    
    n = A.shape[-1]
    if A.shape[-2] != n:
        raise ValueError("Input tensor must be square")
    
    batch_dims = A.shape[:-2]
    batch_size = 1
    for dim in batch_dims:
        batch_size *= dim
    
    # Create output tensor
    if out is None:
        out = torch.empty_like(A)
    else:
        if out.shape != A.shape:
            raise ValueError("Output tensor must have the same shape as input tensor")
    
    # Determine block size
    BLOCK_SIZE = min(32, n)
    
    # Launch kernel
    grid = (batch_size, (n + BLOCK_SIZE - 1) // BLOCK_SIZE)
    cholesky_kernel[grid](
        A,
        out,
        n,
        batch_size,
        A.stride(-2),
        A.stride(-1),
        BLOCK_SIZE
    )
    
    # Handle upper triangular case
    if upper:
        out = out.transpose(-2, -1).conj()
    
    return out
