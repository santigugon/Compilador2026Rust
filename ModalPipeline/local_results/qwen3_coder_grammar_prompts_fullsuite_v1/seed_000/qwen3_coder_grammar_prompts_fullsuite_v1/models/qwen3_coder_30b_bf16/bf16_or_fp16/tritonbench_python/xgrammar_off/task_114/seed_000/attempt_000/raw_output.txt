import torch
import triton
import triton.language as tl
import math

@triton.jit
def _determinant_lu_kernel(A_ptr, P_ptr, out_ptr, n: tl.constexpr, batch_size: tl.constexpr, pivot: tl.constexpr, BLOCK: tl.constexpr):
    batch_idx = tl.program_id(0)
    if batch_idx >= batch_size:
        return
    
    # Each batch gets its own block
    A_base = A_ptr + batch_idx * n * n
    P_base = P_ptr + batch_idx * n
    
    # Initialize determinant to 1
    det = tl.full([1], 1.0, dtype=tl.float64)
    sign = tl.full([1], 1.0, dtype=tl.float64)
    
    # Initialize pivot array
    for i in range(n):
        tl.store(P_base + i, i)
    
    # LU decomposition with partial pivoting
    for k in range(n):
        # Find pivot
        if pivot:
            max_val = tl.abs(tl.load(A_base + k * n + k))
            pivot_row = k
            for i in range(k + 1, n):
                val = tl.abs(tl.load(A_base + i * n + k))
                if val > max_val:
                    max_val = val
                    pivot_row = i
            
            # Swap rows if needed
            if pivot_row != k:
                # Swap rows in A
                for j in range(n):
                    temp = tl.load(A_base + k * n + j)
                    tl.store(A_base + k * n + j, tl.load(A_base + pivot_row * n + j))
                    tl.store(A_base + pivot_row * n + j, temp)
                
                # Update pivot array
                temp_pivot = tl.load(P_base + k)
                tl.store(P_base + k, tl.load(P_base + pivot_row))
                tl.store(P_base + pivot_row, temp_pivot)
                
                # Update sign
                sign = -sign
        
        # Check for zero pivot
        pivot_val = tl.load(A_base + k * n + k)
        if pivot_val == 0.0:
            # Matrix is singular, determinant is 0
            det = tl.full([1], 0.0, dtype=tl.float64)
            break
        
        # Update determinant
        det = det * pivot_val
        
        # Eliminate column
        for i in range(k + 1, n):
            factor = tl.load(A_base + i * n + k) / pivot_val
            tl.store(A_base + i * n + k, factor)
            for j in range(k + 1, n):
                val = tl.load(A_base + i * n + j) - factor * tl.load(A_base + k * n + j)
                tl.store(A_base + i * n + j, val)
    
    # Apply sign to determinant
    det = det * sign
    
    # Store result
    tl.store(out_ptr + batch_idx, det)

def determinant_lu(A, *, pivot=True, out=None):
    # Validate input
    if A.dim() < 2:
        raise ValueError("Input tensor must have at least 2 dimensions")
    
    batch_dims = A.shape[:-2]
    n = A.shape[-2]
    if A.shape[-1] != n:
        raise ValueError("Input tensor must be square")
    
    # Handle batch dimensions
    batch_size = 1
    for dim in batch_dims:
        batch_size *= dim
    
    # Create output tensor
    if out is None:
        out = torch.empty(batch_dims, dtype=torch.float64, device=A.device)
    else:
        if out.shape != batch_dims:
            raise ValueError("Output tensor shape does not match batch dimensions")
        if out.dtype != torch.float64:
            raise ValueError("Output tensor must be of type float64")
    
    # Handle scalar case
    if batch_size == 1:
        # For scalar case, we can use a simpler approach
        if n == 1:
            if out is not None:
                out.fill_(A[0, 0].item())
            else:
                out = torch.tensor(A[0, 0].item(), dtype=torch.float64, device=A.device)
            return out
        elif n == 2:
            # For 2x2 matrix: det = ad - bc
            a = A[0, 0]
            b = A[0, 1]
            c = A[1, 0]
            d = A[1, 1]
            det = a * d - b * c
            if out is not None:
                out.fill_(det.item())
            else:
                out = torch.tensor(det.item(), dtype=torch.float64, device=A.device)
            return out
    
    # For larger matrices, use the full LU decomposition approach
    # Allocate temporary pivot array
    pivot_tensor = torch.empty(batch_size, n, dtype=torch.int32, device=A.device)
    
    # Launch kernel
    block = 256
    grid = (batch_size,)
    
    # Create a contiguous copy of A for the kernel
    A_contiguous = A.contiguous()
    
    # Launch kernel
    _determinant_lu_kernel[grid](
        A_contiguous,
        pivot_tensor,
        out,
        n,
        batch_size,
        pivot,
        BLOCK=block
    )
    
    return out
