import torch
import triton
import triton.language as tl

@triton.jit
def _fused_masked_select_add_gelu_kernel(
    input_ptr, mask_ptr, other_ptr, out_ptr,
    n: tl.constexpr,
    alpha: tl.constexpr,
    other_is_scalar: tl.constexpr,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    
    # Load input values
    input_vals = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    
    # Load mask
    mask_vals = tl.load(mask_ptr + offsets, mask=mask, other=False)
    
    # Select values based on mask
    selected_vals = tl.where(mask_vals, input_vals, 0.0)
    
    # Load other value (scalar or tensor)
    if other_is_scalar:
        other_vals = other_ptr[0] * alpha
    else:
        other_vals = tl.load(other_ptr + offsets, mask=mask, other=0.0)
        other_vals = other_vals * alpha
    
    # Add other to selected values
    added_vals = selected_vals + other_vals
    
    # Apply GELU activation
    # GELU(x) = 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    # Using approximation: 0.5 * x * (1 + tanh(sqrt(2/pi) * x * (1 + 0.044715 * x^2)))
    sqrt_2_over_pi = 0.7978845608028654  # sqrt(2/pi)
    coeff = 0.044715
    
    # Compute GELU
    x = added_vals
    x_cubed = x * x * x
    inner = sqrt_2_over_pi * x * (1.0 + coeff * x_cubed)
    tanh_val = 2.0 / (1.0 + tl.exp(-2.0 * inner)) - 1.0
    gelu_val = 0.5 * x * (1.0 + tanh_val)
    
    # Store result
    tl.store(out_ptr + offsets, gelu_val, mask=mask)

def fused_masked_select_add_gelu(input, mask, other, *, alpha=1, approximate='none', out=None):
    # Validate inputs
    if input.shape != mask.shape:
        raise ValueError("input and mask must have the same shape")
    
    # Handle scalar other
    other_is_scalar = not torch.is_tensor(other)
    if other_is_scalar:
        other_tensor = torch.tensor(other, dtype=input.dtype, device=input.device)
    else:
        if input.shape != other.shape:
            raise ValueError("input and other must have the same shape when other is a tensor")
        other_tensor = other
    
    # Prepare output
    if out is None:
        out = torch.empty_like(input)
    else:
        if out.shape != input.shape:
            raise ValueError("out must have the same shape as input")
    
    # Flatten tensors for kernel processing
    n = input.numel()
    
    # Launch kernel
    block = 256
    grid = (triton.cdiv(n, block),)
    
    _fused_masked_select_add_gelu_kernel[grid](
        input, mask, other_tensor, out,
        n,
        alpha,
        other_is_scalar,
        BLOCK=block
    )
    
    return out
