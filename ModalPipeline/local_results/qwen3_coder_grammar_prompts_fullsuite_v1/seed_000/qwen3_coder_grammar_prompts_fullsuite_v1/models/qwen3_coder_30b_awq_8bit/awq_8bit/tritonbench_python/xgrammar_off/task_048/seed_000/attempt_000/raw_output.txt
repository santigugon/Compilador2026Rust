```python
import torch
import triton
import triton.language as tl

@triton.jit
def _scaled_add_norm_kernel(y_ptr, x_ptr, out_ptr, n: tl.constexpr, alpha: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    
    # Load y and x
    y = tl.load(y_ptr + offsets, mask=mask, other=0.0)
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Compute y += alpha * x
    y_new = y + alpha * x
    tl.store(y_ptr + offsets, y_new, mask=mask)
    
    # Compute squared values for norm
    y_squared = y_new * y_new
    tl.store(out_ptr + offsets, y_squared, mask=mask)

def scaled_add_norm(y, x, alpha):
    # Create output tensor for squared values
    out_squared = torch.empty_like(y)
    
    # Compute the 2-norm using Triton kernel
    n = y.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    
    # Launch kernel to compute y += alpha * x and store squared values
    _scaled_add_norm_kernel[grid](y, x, out_squared, n, alpha, BLOCK=block)
    
    # Compute the 2-norm from squared values
    norm = torch.sqrt(torch.sum(out_squared))
    
    return norm
```
