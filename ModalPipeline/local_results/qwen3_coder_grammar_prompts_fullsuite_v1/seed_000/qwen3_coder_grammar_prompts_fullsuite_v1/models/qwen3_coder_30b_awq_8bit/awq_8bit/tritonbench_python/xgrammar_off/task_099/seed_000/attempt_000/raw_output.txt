import torch
import triton
import triton.language as tl

@triton.jit
def _gelu_kernel(x_ptr, out_ptr, n: tl.constexpr, BLOCK: tl.constexpr, approximate: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    if approximate == 'none':
        # Standard GELU: x * 0.5 * (1 + erf(x / sqrt(2)))
        sqrt_2 = 1.4142135623730951
        erf_arg = x / sqrt_2
        # Approximate erf using polynomial approximation
        # erf(x) ≈ sign(x) * (1 - exp(-x^2 * (4/π + a*x^2)/(1 + a*x^2)))
        a = 0.147
        exp_arg = -erf_arg * erf_arg * (4.0 / 3.141592653589793 + a * erf_arg * erf_arg)
        erf_val = erf_arg * (1.0 - tl.exp(exp_arg))
        y = x * 0.5 * (1.0 + erf_val)
    else:
        # Approximate GELU using tanh
        # GELU ≈ 0.5 * x * (1 + tanh(sqrt(2/π) * (x + 0.044715 * x^3)))
        sqrt_2_over_pi = 0.7978845608028654
        x_cubed = x * x * x
        tanh_arg = sqrt_2_over_pi * (x + 0.044715 * x_cubed)
        y = 0.5 * x * (1.0 + tl.tanh(tanh_arg))
    
    tl.store(out_ptr + offsets, y, mask=mask)

@triton.jit
def _std_kernel(x_ptr, out_ptr, mean_ptr, n: tl.constexpr, reduced_n: tl.constexpr, BLOCK: tl.constexpr, keepdim: tl.constexpr, correction: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Load mean
    mean = tl.load(mean_ptr + pid, mask=pid < reduced_n, other=0.0)
    
    # Compute squared differences
    diff = x - mean
    squared_diff = diff * diff
    
    # Store squared differences for reduction
    tl.store(out_ptr + offsets, squared_diff, mask=mask)

def gelu_std(input, dim=None, keepdim=False, correction=1, approximate='none', out=None):
    # Handle scalar input
    if input.dim() == 0:
        input = input.unsqueeze(0)
        if dim is not None:
            dim = dim if isinstance(dim, tuple) else (dim,)
            dim = tuple(d + 1 if d >= 0 else d for d in dim)
    
    # Apply GELU activation
    gelu_input = input
    if approximate not in ['none', 'tanh']:
        approximate = 'none'
    
    # Create output tensor for GELU
    gelu_out = torch.empty_like(input)
    
    # Launch GELU kernel
    n = input.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    _gelu_kernel[grid](input, gelu_out, n, BLOCK=block, approximate=approximate)
    
    # Compute mean for standard deviation calculation
    if dim is None:
        # Reduce all dimensions
        reduced_shape = []
        reduced_dims = list(range(input.dim()))
    else:
        reduced_dims = dim if isinstance(dim, tuple) else (dim,)
        # Normalize negative dimensions
        reduced_dims = tuple(d if d >= 0 else d + input.dim() for d in reduced_dims)
        # Ensure no duplicates
        reduced_dims = tuple(sorted(set(reduced_dims)))
        # Compute reduced shape
        reduced_shape = [input.shape[i] for i in range(input.dim()) if i not in reduced_dims]
    
    # Compute mean
    if len(reduced_dims) == 0:
        # No reduction needed
        mean = gelu_out.mean()
    else:
        # Compute mean along specified dimensions
        mean = gelu_out.mean(dim=reduced_dims, keepdim=True)
    
    # Compute standard deviation
    if len(reduced_dims) == 0:
        # All elements reduced
        if correction == 0:
            std = torch.sqrt(torch.mean((gelu_out - mean) ** 2))
        else:
            std = torch.sqrt(torch.var(gelu_out, correction=correction))
    else:
        # Partial reduction
        # Compute squared differences
        squared_diff = (gelu_out - mean) ** 2
        # Sum along reduced dimensions
        sum_squared_diff = squared_diff.sum(dim=reduced_dims, keepdim=True)
        # Compute variance
        if correction == 0:
            var = sum_squared_diff / torch.prod(torch.tensor([input.shape[i] for i in reduced_dims]))
        else:
            var = sum_squared_diff / (torch.prod(torch.tensor([input.shape[i] for i in reduced_dims])) - correction)
        # Standard deviation
        std = torch.sqrt(var)
    
    # Handle keepdim
    if keepdim and len(reduced_dims) > 0:
        # Expand dimensions to match keepdim
        if len(reduced_shape) > 0:
            std = std.expand(input.shape)
        else:
            std = std.expand(input.shape)
    
    # Handle output tensor
    if out is not None:
        out.copy_(std)
        return out
    else:
        return std
