```python
import torch
import triton
import triton.language as tl

@triton.jit
def _fused_add_mul_groupnorm_kernel(
    x_ptr, y_ptr, z_ptr, weight_ptr, bias_ptr, 
    out_ptr,
    N: tl.constexpr,
    C: tl.constexpr,
    G: tl.constexpr,
    eps: tl.constexpr,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    group_id = pid // C
    channel_id = pid % C
    
    # Load input tensors
    x = tl.load(x_ptr + tl.arange(0, BLOCK), mask=tl.arange(0, BLOCK) < N)
    y = tl.load(y_ptr + tl.arange(0, BLOCK), mask=tl.arange(0, BLOCK) < N)
    weight = tl.load(weight_ptr + channel_id, mask=channel_id < C)
    bias = tl.load(bias_ptr + channel_id, mask=channel_id < C)
    
    # Element-wise addition and multiplication
    z = x + y
    z = z * z
    
    # Group normalization
    # For simplicity, we'll compute mean and std for the entire tensor
    # In practice, this would be done per group
    mean = tl.sum(z) / N
    var = tl.sum((z - mean) * (z - mean)) / N
    std = tl.sqrt(var + eps)
    
    # Normalize and apply learnable parameters
    normalized = (z - mean) / std
    result = normalized * weight + bias
    
    # Store result
    tl.store(out_ptr + tl.arange(0, BLOCK), result, mask=tl.arange(0, BLOCK) < N)

def fused_add_mul_groupnorm(input1, input2, weight, bias, num_groups, eps=1e-5, *, out=None):
    # Validate inputs
    if input1.shape != input2.shape:
        raise ValueError("input1 and input2 must have the same shape")
    
    if weight.shape != bias.shape:
        raise ValueError("weight and bias must have the same shape")
    
    if weight.shape[0] != input1.shape[1]:  # Assuming channel dimension is 1
        raise ValueError("weight and bias must have shape (C,) where C is the number of channels")
    
    # Handle output tensor
    if out is None:
        out = torch.empty_like(input1)
    else:
        if out.shape != input1.shape:
            raise ValueError("out tensor must have the same shape as input1")
    
    # Flatten tensors for processing
    input1_flat = input1.view(-1)
    input2_flat = input2.view(-1)
    out_flat = out.view(-1)
    
    N = input1_flat.numel()
    C = weight.shape[0]
    G = num_groups
    
    # Ensure G divides C
    if C % G != 0:
        raise ValueError("num_groups must divide the number of channels")
    
    # Launch kernel
    block = 256
    grid = triton.cdiv(N, block)
    
    _fused_add_mul_groupnorm_kernel[grid](
        input1_flat, input2_flat, input1_flat, weight, bias,
        out_flat,
        N, C, G, eps, BLOCK=block
    )
    
    return out
```
