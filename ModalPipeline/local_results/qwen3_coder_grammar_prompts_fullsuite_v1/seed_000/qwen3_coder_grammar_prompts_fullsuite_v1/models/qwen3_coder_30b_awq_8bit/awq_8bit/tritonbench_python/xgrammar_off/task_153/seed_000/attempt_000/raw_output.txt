import torch
import triton
import triton.language as tl

@triton.jit
def adaptive_avg_pool2d_kernel(
    input_ptr, output_ptr,
    input_shape_0, input_shape_1, input_shape_2, input_shape_3,
    output_shape_0, output_shape_1, output_shape_2, output_shape_3,
    BLOCK_SIZE_H: tl.constexpr, BLOCK_SIZE_W: tl.constexpr
):
    # Get the block index
    block_id_h = tl.program_id(0)
    block_id_w = tl.program_id(1)
    
    # Get the output indices
    output_h = block_id_h * BLOCK_SIZE_H
    output_w = block_id_w * BLOCK_SIZE_W
    
    # Get the input dimensions
    input_h = input_shape_2
    input_w = input_shape_3
    
    # Get the output dimensions
    output_h_size = output_shape_2
    output_w_size = output_shape_3
    
    # Calculate the start and end indices for the input
    start_h = (output_h * input_h) // output_h_size
    end_h = ((output_h + BLOCK_SIZE_H) * input_h + output_h_size - 1) // output_h_size
    start_w = (output_w * input_w) // output_w_size
    end_w = ((output_w + BLOCK_SIZE_W) * input_w + output_w_size - 1) // output_w_size
    
    # Calculate the number of elements in the pooling region
    num_elements = (end_h - start_h) * (end_w - start_w)
    
    # Initialize the sum
    sum_val = 0.0
    
    # Iterate over the pooling region
    for h in range(start_h, end_h):
        for w in range(start_w, end_w):
            # Calculate the input index
            input_idx = h * input_w + w
            # Add the value to the sum
            sum_val += tl.load(input_ptr + input_idx)
    
    # Calculate the average
    avg_val = sum_val / num_elements
    
    # Store the result
    output_idx = output_h * output_w_size + output_w
    tl.store(output_ptr + output_idx, avg_val)

def adaptive_avg_pool2d(output_size):
    def wrapper(input_tensor):
        # Determine input shape
        if input_tensor.dim() == 3:
            N, C, H_in, W_in = 1, input_tensor.shape[0], input_tensor.shape[1], input_tensor.shape[2]
        else:
            N, C, H_in, W_in = input_tensor.shape
            
        # Handle output size
        if isinstance(output_size, int):
            output_h, output_w = output_size, output_size
        else:
            output_h, output_w = output_size
            
        # Create output tensor
        output = torch.empty((N, C, output_h, output_w), device=input_tensor.device, dtype=input_tensor.dtype)
        
        # Launch kernel
        grid = (
            triton.cdiv(output_h, 16),
            triton.cdiv(output_w, 16)
        )
        
        adaptive_avg_pool2d_kernel[grid](
            input_tensor,
            output,
            N, C, H_in, W_in,
            N, C, output_h, output_w,
            BLOCK_SIZE_H=16,
            BLOCK_SIZE_W=16
        )
        
        return output
    
    return wrapper
