import torch
import triton
import triton.language as tl

@triton.jit
def _log_softmax_linear_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_row_stride, weight_row_stride, weight_col_stride,
    output_row_stride, output_col_stride,
    input_size: tl.constexpr,
    weight_size: tl.constexpr,
    output_size: tl.constexpr,
    BLOCK_SIZE: tl.constexpr
):
    # Get the row index
    row_idx = tl.program_id(0)
    
    # Load input row
    input_offsets = row_idx * input_row_stride + tl.arange(0, BLOCK_SIZE)
    input_mask = tl.arange(0, BLOCK_SIZE) < input_size
    input_row = tl.load(input_ptr + input_offsets, mask=input_mask, other=0.0)
    
    # Compute linear transformation: input @ weight.T + bias
    output_offsets = row_idx * output_row_stride + tl.arange(0, BLOCK_SIZE)
    output_mask = tl.arange(0, BLOCK_SIZE) < output_size
    
    # Initialize output with bias
    if bias_ptr is not None:
        bias_offsets = tl.arange(0, BLOCK_SIZE)
        bias_mask = tl.arange(0, BLOCK_SIZE) < output_size
        output = tl.load(bias_ptr + bias_offsets, mask=bias_mask, other=0.0)
    else:
        output = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    
    # Compute matrix multiplication
    for i in range(0, input_size, BLOCK_SIZE):
        # Load weight column
        weight_offsets = i * weight_row_stride + tl.arange(0, BLOCK_SIZE)
        weight_mask = (tl.arange(0, BLOCK_SIZE) + i) < input_size
        weight_col = tl.load(weight_ptr + weight_offsets, mask=weight_mask, other=0.0)
        
        # Compute dot product
        dot_product = tl.sum(input_row * weight_col)
        output += dot_product
    
    # Apply log_softmax
    # For numerical stability, subtract max value
    max_val = tl.max(output, axis=0)
    exp_output = tl.exp(output - max_val)
    sum_exp = tl.sum(exp_output, axis=0)
    log_softmax_output = (output - max_val) - tl.log(sum_exp)
    
    # Store result
    tl.store(output_ptr + output_offsets, log_softmax_output, mask=output_mask)
