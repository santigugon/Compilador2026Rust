import torch
import triton
import triton.language as tl

@triton.jit
def _pixel_shuffle_conv2d_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_height, input_width, output_height, output_width,
    in_channels, out_channels, kernel_height, kernel_width,
    stride_h, stride_w, padding_h, padding_w, dilation_h, dilation_w,
    groups, upscale_factor,
    BLOCK_SIZE: tl.constexpr
):
    # Get program ID
    pid = tl.program_id(0)
    
    # Calculate output dimensions
    batch_size = input_height // upscale_factor
    spatial_size = output_height * output_width
    
    # Each program processes one batch element
    batch_idx = pid // spatial_size
    spatial_idx = pid % spatial_size
    
    if batch_idx >= batch_size:
        return
    
    # Calculate spatial coordinates
    out_h = spatial_idx // output_width
    out_w = spatial_idx % output_width
    
    # Calculate input coordinates after upsampling
    in_h = out_h // upscale_factor
    in_w = out_w // upscale_factor
    
    # Calculate group information
    group_size = in_channels // groups
    out_group_size = out_channels // groups
    
    # Initialize output value
    output_val = 0.0
    
    # Perform convolution for each group
    for g in range(groups):
        # Calculate group offsets
        group_in_start = g * group_size
        group_out_start = g * out_group_size
        
        # Convolution loop
        for kh in range(kernel_height):
            for kw in range(kernel_width):
                # Calculate input coordinates
                ih = in_h * stride_h - padding_h + kh * dilation_h
                iw = in_w * stride_w - padding_w + kw * dilation_w
                
                # Check bounds
                if ih >= 0 and ih < input_height and iw >= 0 and iw < input_width:
                    # Calculate input index
                    input_idx = batch_idx * (in_channels * input_height * input_width) + \
                               group_in_start * (input_height * input_width) + \
                               ih * input_width + iw
                    
                    # Calculate weight index
                    weight_idx = group_out_start * (group_size * kernel_height * kernel_width) + \
                                0 * (group_size * kernel_height * kernel_width) + \
                                kh * (group_size * kernel_width) + \
                                kw * group_size
                    
                    # Load input and weight
                    input_val = tl.load(input_ptr + input_idx, mask=True)
                    weight_val = tl.load(weight_ptr + weight_idx, mask=True)
                    
                    # Accumulate
                    output_val += input_val * weight_val
    
    # Add bias if provided
    if bias_ptr is not None:
        # Calculate bias index
        bias_idx = 0  # Simplified - assuming bias is broadcast
        bias_val = tl.load(bias_ptr + bias_idx, mask=True)
        output_val += bias_val
    
    # Calculate output index
    output_idx = batch_idx * (out_channels * output_height * output_width) + \
                0 * (output_height * output_width) + \
                out_h * output_width + out_w
    
    # Store result
    tl.store(output_ptr + output_idx, output_val, mask=True)

def pixel_shuffle_conv2d(
    input: torch.Tensor, 
    weight: torch.Tensor, 
    bias=None, 
    stride=1, 
    padding=0, 
    dilation=1, 
    groups=1, 
    upscale_factor=2
) -> torch.Tensor:
    # Get input dimensions
    batch_size, in_channels, iH, iW = input.shape
    out_channels, _, kH, kW = weight.shape
    
    # Calculate output dimensions
    output_height = (iH + 2 * padding - (dilation * (kH - 1) + 1)) // stride + 1
    output_width = (iW + 2 * padding - (dilation * (kW - 1) + 1)) // stride + 1
    
    # Apply upscale factor to output dimensions
    output_height *= upscale_factor
    output_width *= upscale_factor
    
    # Create output tensor
    output = torch.empty(
        batch_size, out_channels, output_height, output_width,
        dtype=input.dtype, device=input.device
    )
    
    # Handle bias
    if bias is not None:
        bias = bias.to(input.dtype)
    
    # Convert stride, padding, dilation to tuples if needed
    if not isinstance(stride, tuple):
        stride = (stride, stride)
    if not isinstance(padding, tuple):
        padding = (padding, padding)
    if not isinstance(dilation, tuple):
        dilation = (dilation, dilation)
    
    stride_h, stride_w = stride
    padding_h, padding_w = padding
    dilation_h, dilation_w = dilation
    
    # Launch kernel
    block_size = 256
    grid_size = batch_size * output_height * output_width
    
    # Create kernel arguments
    kernel_args = {
        'input_ptr': input.data_ptr(),
        'weight_ptr': weight.data_ptr(),
        'bias_ptr': bias.data_ptr() if bias is not None else None,
        'output_ptr': output.data_ptr(),
        'input_height': iH,
        'input_width': iW,
        'output_height': output_height,
        'output_width': output_width,
        'in_channels': in_channels,
        'out_channels': out_channels,
        'kernel_height': kH,
        'kernel_width': kW,
        'stride_h': stride_h,
        'stride_w': stride_w,
        'padding_h': padding_h,
        'padding_w': padding_w,
        'dilation_h': dilation_h,
        'dilation_w': dilation_w,
        'groups': groups,
        'upscale_factor': upscale_factor,
        'BLOCK_SIZE': block_size
    }
    
    # Launch kernel
    _pixel_shuffle_conv2d_kernel[grid_size](**kernel_args)
    
    return output
