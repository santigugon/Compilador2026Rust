import torch
import triton
import triton.language as tl
import math

@triton.jit
def _lu_decompose_kernel(A_ptr, L_ptr, U_ptr, pivot_ptr, n, batch_size, BLOCK: tl.constexpr):
    # For each batch
    batch_idx = tl.program_id(0)
    # For each row in the matrix
    row = tl.program_id(1)
    
    # Load matrix A for this batch
    A_batch = A_ptr + batch_idx * n * n
    L_batch = L_ptr + batch_idx * n * n
    U_batch = U_ptr + batch_idx * n * n
    pivot_batch = pivot_ptr + batch_idx * n
    
    # Initialize L and U matrices
    for i in range(n):
        for j in range(n):
            if i == j:
                tl.store(L_batch + i * n + j, 1.0)
            else:
                tl.store(L_batch + i * n + j, 0.0)
    
    # Initialize U with A values
    for i in range(n):
        for j in range(n):
            tl.store(U_batch + i * n + j, tl.load(A_batch + i * n + j))
    
    # LU decomposition
    for k in range(n):
        # Find pivot
        max_val = tl.abs(tl.load(U_batch + k * n + k))
        pivot_idx = k
        for i in range(k + 1, n):
            val = tl.abs(tl.load(U_batch + i * n + k))
            if val > max_val:
                max_val = val
                pivot_idx = i
        
        # Swap rows in U
        if pivot_idx != k:
            for j in range(n):
                temp = tl.load(U_batch + k * n + j)
                tl.store(U_batch + k * n + j, tl.load(U_batch + pivot_idx * n + j))
                tl.store(U_batch + pivot_idx * n + j, temp)
        
        # Store pivot
        tl.store(pivot_batch + k, pivot_idx)
        
        # Compute L and U
        for i in range(k + 1, n):
            if tl.load(U_batch + k * n + k) != 0.0:
                factor = tl.load(U_batch + i * n + k) / tl.load(U_batch + k * n + k)
                tl.store(L_batch + i * n + k, factor)
                for j in range(k, n):
                    val = tl.load(U_batch + i * n + j) - factor * tl.load(U_batch + k * n + j)
                    tl.store(U_batch + i * n + j, val)
            else:
                tl.store(L_batch + i * n + k, 0.0)

@triton.jit
def _solve_triangular_kernel(L_ptr, U_ptr, pivot_ptr, b_ptr, x_ptr, n, batch_size, BLOCK: tl.constexpr):
    # For each batch
    batch_idx = tl.program_id(0)
    
    # Load L, U, pivot, and b for this batch
    L_batch = L_ptr + batch_idx * n * n
    U_batch = U_ptr + batch_idx * n * n
    pivot_batch = pivot_ptr + batch_idx * n
    b_batch = b_ptr + batch_idx * n
    x_batch = x_ptr + batch_idx * n
    
    # Create a copy of b for solving
    y = tl.zeros((n,), dtype=tl.float32)
    for i in range(n):
        tl.store(y + i, tl.load(b_batch + i))
    
    # Forward substitution with pivoting
    for i in range(n):
        pivot_i = tl.load(pivot_batch + i)
        if pivot_i != i:
            temp = tl.load(y + i)
            tl.store(y + i, tl.load(y + pivot_i))
            tl.store(y + pivot_i, temp)
        
        for j in range(i):
            val = tl.load(y + i) - tl.load(L_batch + i * n + j) * tl.load(y + j)
            tl.store(y + i, val)
    
    # Backward substitution
    for i in range(n - 1, -1, -1):
        for j in range(i + 1, n):
            val = tl.load(y + i) - tl.load(U_batch + i * n + j) * tl.load(x_batch + j)
            tl.store(y + i, val)
        
        if tl.load(U_batch + i * n + i) != 0.0:
            val = tl.load(y + i) / tl.load(U_batch + i * n + i)
            tl.store(x_batch + i, val)
        else:
            tl.store(x_batch + i, 0.0)

def invert_matrix_lu(A, *, pivot=True, out=None):
    if not torch.is_tensor(A):
        raise TypeError("Input must be a tensor")
    
    if A.dim() < 2:
        raise ValueError("Input must be at least 2D")
    
    if A.size(-1) != A.size(-2):
        raise ValueError("Input must be square matrices")
    
    batch_dims = A.shape[:-2]
    n = A.size(-1)
    
    # Create output tensor
    if out is not None:
        if out.shape != A.shape:
            raise ValueError("Output tensor must have the same shape as input")
        if out.dtype != A.dtype:
            raise ValueError("Output tensor must have the same dtype as input")
        out = out
    else:
        out = torch.empty_like(A)
    
    # Handle scalar case
    if A.numel() == 1:
        out.fill_(1.0 / A.item())
        return out
    
    # Handle batched case
    if len(batch_dims) > 0:
        batch_size = math.prod(batch_dims)
    else:
        batch_size = 1
    
    # Allocate intermediate tensors
    L = torch.empty_like(A)
    U = torch.empty_like(A)
    pivot = torch.empty((batch_size, n), dtype=torch.int32, device=A.device)
    
    # Launch LU decomposition kernel
    if batch_size > 1:
        grid = (batch_size, n)
    else:
        grid = (1, n)
    
    block = 16
    _lu_decompose_kernel[grid](A, L, U, pivot, n, batch_size, BLOCK=block)
    
    # Create identity matrix for solving
    identity = torch.eye(n, dtype=A.dtype, device=A.device)
    identity = identity.expand(batch_size, n, n)
    
    # Solve for each column of the identity matrix
    x = torch.empty_like(identity)
    
    # Launch triangular solve kernel
    _solve_triangular_kernel[grid](L, U, pivot, identity, x, n, batch_size, BLOCK=block)
    
    # Transpose to get inverse
    if batch_size > 1:
        out = x.transpose(-1, -2).reshape(A.shape)
    else:
        out = x.transpose(-1, -2)
    
    return out
