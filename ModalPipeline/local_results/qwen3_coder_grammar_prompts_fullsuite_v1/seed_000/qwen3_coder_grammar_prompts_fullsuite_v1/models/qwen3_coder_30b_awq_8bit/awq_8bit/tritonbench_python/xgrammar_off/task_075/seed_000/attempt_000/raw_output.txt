import torch
import triton
import triton.language as tl

@triton.jit
def _cholesky_kernel(A_ptr, L_ptr, n: tl.constexpr, BLOCK: tl.constexpr):
    # Compute Cholesky decomposition: A = L * L.T
    # L is lower triangular
    
    pid = tl.program_id(0)
    block_size = BLOCK
    
    # Each thread block handles one row of L
    row = pid
    
    if row >= n:
        return
    
    # For each element in the row
    for col in range(row + 1):
        # Compute the dot product of the row of L and column of L
        sum_val = 0.0
        for k in range(col):
            sum_val += L_ptr[row * n + k] * L_ptr[col * n + k]
        
        if row == col:
            # Diagonal element: sqrt(A[row, col] - sum_val)
            val = A_ptr[row * n + col] - sum_val
            L_ptr[row * n + col] = tl.sqrt(val)
        else:
            # Off-diagonal element: (A[row, col] - sum_val) / L[col, col]
            val = A_ptr[row * n + col] - sum_val
            L_ptr[row * n + col] = val / L_ptr[col * n + col]

@triton.jit
def _forward_substitution_kernel(L_ptr, b_ptr, x_ptr, n: tl.constexpr, k: tl.constexpr, BLOCK: tl.constexpr):
    # Solve L * x = b using forward substitution
    # Each thread block handles one column of x
    
    pid = tl.program_id(0)
    col = pid
    
    if col >= k:
        return
    
    # For each row
    for row in range(n):
        sum_val = 0.0
        for j in range(row):
            sum_val += L_ptr[row * n + j] * x_ptr[j * k + col]
        
        x_ptr[row * k + col] = (b_ptr[row * k + col] - sum_val) / L_ptr[row * n + row]

@triton.jit
def _backward_substitution_kernel(L_ptr, x_ptr, n: tl.constexpr, k: tl.constexpr, BLOCK: tl.constexpr):
    # Solve L.T * y = x using backward substitution
    # Each thread block handles one column of y
    
    pid = tl.program_id(0)
    col = pid
    
    if col >= k:
        return
    
    # For each row from bottom to top
    for row in range(n - 1, -1, -1):
        sum_val = 0.0
        for j in range(row + 1, n):
            sum_val += L_ptr[j * n + row] * x_ptr[j * k + col]
        
        x_ptr[row * k + col] = (x_ptr[row * k + col] - sum_val) / L_ptr[row * n + row]

def fused_cholesky_solve(A: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
    # Input validation
    assert A.dim() == 2 and A.size(0) == A.size(1), "A must be a square matrix"
    assert b.dim() == 2, "b must be a 2D tensor"
    assert A.size(0) == b.size(0), "A and b must have compatible dimensions"
    
    n, k = A.size(0), b.size(1)
    
    # Allocate output tensor
    x = torch.empty_like(b)
    
    # Copy b to x for the forward substitution step
    x.copy_(b)
    
    # Allocate memory for L (lower triangular matrix)
    L = torch.empty_like(A)
    
    # Compute Cholesky decomposition
    block = 16
    grid = (triton.cdiv(n, block),)
    _cholesky_kernel[grid](A, L, n, BLOCK=block)
    
    # Forward substitution: L * y = b
    grid = (triton.cdiv(k, block),)
    _forward_substitution_kernel[grid](L, b, x, n, k, BLOCK=block)
    
    # Backward substitution: L.T * x = y
    grid = (triton.cdiv(k, block),)
    _backward_substitution_kernel[grid](L, x, n, k, BLOCK=block)
    
    return x
