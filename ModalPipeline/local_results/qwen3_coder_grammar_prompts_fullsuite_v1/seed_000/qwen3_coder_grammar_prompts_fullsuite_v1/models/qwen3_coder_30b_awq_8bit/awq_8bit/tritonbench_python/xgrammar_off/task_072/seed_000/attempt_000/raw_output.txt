import torch
import triton
import triton.language as tl

@triton.jit
def _conv2d_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    input_strides, weight_strides, output_strides,
    batch_size, in_channels, out_channels,
    iH, iW, oH, oW, kH, kW,
    stride_h, stride_w, padding_h, padding_w,
    dilation_h, dilation_w,
    groups, group_size,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    batch_id = pid // (out_channels * oH * oW)
    oc_id = (pid // (oH * oW)) % out_channels
    oh_id = (pid // oW) % oH
    ow_id = pid % oW
    
    if batch_id >= batch_size or oc_id >= out_channels or oh_id >= oH or ow_id >= oW:
        return
    
    # Initialize accumulator
    acc = tl.zeros((1,), dtype=tl.float32)
    
    # Loop over groups and input channels
    for g in range(groups):
        for ic in range(group_size):
            # Calculate input channel for this group
            input_ch = g * group_size + ic
            
            # Loop over kernel
            for kh in range(kH):
                for kw in range(kW):
                    # Calculate input position
                    ih = oh_id * stride_h - padding_h + kh * dilation_h
                    iw = ow_id * stride_w - padding_w + kw * dilation_w
                    
                    # Check bounds
                    if ih >= 0 and ih < iH and iw >= 0 and iw < iW:
                        # Load input and weight
                        input_val = tl.load(input_ptr + 
                                          batch_id * input_strides[0] + 
                                          input_ch * input_strides[1] + 
                                          ih * input_strides[2] + 
                                          iw * input_strides[3])
                        weight_val = tl.load(weight_ptr + 
                                           oc_id * weight_strides[0] + 
                                           ic * weight_strides[1] + 
                                           kh * weight_strides[2] + 
                                           kw * weight_strides[3])
                        acc += input_val * weight_val
    
    # Add bias if present
    if bias_ptr is not None:
        bias_val = tl.load(bias_ptr + oc_id)
        acc += bias_val
    
    # Store result
    tl.store(output_ptr + 
             batch_id * output_strides[0] + 
             oc_id * output_strides[1] + 
             oh_id * output_strides[2] + 
             ow_id * output_strides[3], 
             acc)

@triton.jit
def _batch_norm_kernel(
    input_ptr, output_ptr, mean_ptr, var_ptr, weight_ptr, bias_ptr,
    batch_size, channels, height, width,
    input_strides, output_strides,
    eps: tl.constexpr,
    training: tl.constexpr,
    momentum: tl.constexpr,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    batch_id = pid // (channels * height * width)
    ch_id = (pid // (height * width)) % channels
    h_id = (pid // width) % height
    w_id = pid % width
    
    if batch_id >= batch_size or ch_id >= channels or h_id >= height or w_id >= width:
        return
    
    # Load input value
    input_val = tl.load(input_ptr + 
                       batch_id * input_strides[0] + 
                       ch_id * input_strides[1] + 
                       h_id * input_strides[2] + 
                       w_id * input_strides[3])
    
    # Get statistics
    mean_val = tl.load(mean_ptr + ch_id)
    var_val = tl.load(var_ptr + ch_id)
    
    # Normalize
    normalized = (input_val - mean_val) / tl.sqrt(var_val + eps)
    
    # Apply batch norm transformation
    if weight_ptr is not None and bias_ptr is not None:
        weight_val = tl.load(weight_ptr + ch_id)
        bias_val = tl.load(bias_ptr + ch_id)
        output_val = weight_val * normalized + bias_val
    else:
        output_val = normalized
    
    # Store result
    tl.store(output_ptr + 
             batch_id * output_strides[0] + 
             ch_id * output_strides[1] + 
             h_id * output_strides[2] + 
             w_id * output_strides[3], 
             output_val)

@triton.jit
def _relu_kernel(input_ptr, output_ptr, n: tl.constexpr, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n
    x = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    y = tl.maximum(x, 0.0)
    tl.store(output_ptr + offsets, y, mask=mask)

def relu_batch_norm_conv2d(
    input, weight, bias=None, stride=1, padding=0, dilation=1, groups=1,
    running_mean=None, running_var=None, bn_weight=None, bn_bias=None,
    training=False, momentum=0.1, eps=1e-5, inplace=False
):
    # Handle scalar arguments
    if not isinstance(stride, tuple):
        stride = (stride, stride)
    if not isinstance(padding, tuple):
        padding = (padding, padding)
    if not isinstance(dilation, tuple):
        dilation = (dilation, dilation)
    
    stride_h, stride_w = stride
    padding_h, padding_w = padding
    dilation_h, dilation_w = dilation
    
    # Get dimensions
    batch_size, in_channels, iH, iW = input.shape
    out_channels, _, kH, kW = weight.shape
    
    # Calculate output dimensions
    oH = (iH + 2 * padding_h - (dilation_h * (kH - 1) + 1)) // stride_h + 1
    oW = (iW + 2 * padding_w - (dilation_w * (kW - 1) + 1)) // stride_w + 1
    
    # Calculate group size
    group_size = in_channels // groups
    
    # Allocate output tensor
    output = torch.empty(batch_size, out_channels, oH, oW, device=input.device, dtype=input.dtype)
    
    # Convolution step
    conv_output = torch.empty_like(output)
    
    # Setup convolution kernel launch parameters
    grid_size = batch_size * out_channels * oH * oW
    block_size = 256
    grid = (triton.cdiv(grid_size, block_size),)
    
    # Convolution kernel
    _conv2d_kernel[grid](
        input, weight, bias, conv_output,
        input.stride(), weight.stride(), conv_output.stride(),
        batch_size, in_channels, out_channels,
        iH, iW, oH, oW, kH, kW,
        stride_h, stride_w, padding_h, padding_w,
        dilation_h, dilation_w,
        groups, group_size,
        BLOCK_SIZE=block_size
    )
    
    # Batch normalization step
    if training:
        # Compute mean and variance
        mean = conv_output.mean(dim=(0, 2, 3))
        var = conv_output.var(dim=(0, 2, 3), unbiased=False)
        
        # Update running statistics
        if running_mean is not None and running_var is not None:
            running_mean = (1 - momentum) * running_mean + momentum * mean
            running_var = (1 - momentum) * running_var + momentum * var
        
        # Apply batch normalization
        bn_output = torch.empty_like(conv_output)
        _batch_norm_kernel[grid](
            conv_output, bn_output, mean, var, bn_weight, bn_bias,
            batch_size, out_channels, oH, oW,
            conv_output.stride(), bn_output.stride(),
            eps, training, momentum, BLOCK_SIZE=block_size
        )
    else:
        # Use running statistics
        if running_mean is not None and running_var is not None:
            mean = running_mean
            var = running_var
        else:
            # Default to zero mean and unit variance
            mean = torch.zeros(out_channels, device=conv_output.device, dtype=conv_output.dtype)
            var = torch.ones(out_channels, device=conv_output.device, dtype=conv_output.dtype)
        
        # Apply batch normalization
        bn_output = torch.empty_like(conv_output)
        _batch_norm_kernel[grid](
            conv_output, bn_output, mean, var, bn_weight, bn_bias,
            batch_size, out_channels, oH, oW,
            conv_output.stride(), bn_output.stride(),
            eps, training, momentum, BLOCK_SIZE=block_size
        )
    
    # ReLU activation
    relu_output = torch.empty_like(bn_output)
    n = bn_output.numel()
    block = 256
    grid = (triton.cdiv(n, block),)
    _relu_kernel[grid](bn_output, relu_output, n, BLOCK=block)
    
    return relu_output
